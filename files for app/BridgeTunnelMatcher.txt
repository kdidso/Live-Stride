package com.kdidso.livestride;

import java.util.ArrayDeque;
import java.util.ArrayList;
import java.util.Collections;
import java.util.Comparator;
import java.util.Deque;
import java.util.HashSet;
import java.util.List;
import java.util.Set;

final class BridgeTunnelMatcher {
    static final double ENTRY_RADIUS_METERS = 7.0;
    static final double STRICT_ENTRY_RADIUS_METERS = 5.0;
    static final double CONNECTION_RADIUS_METERS = 4.0;
    static final double STRONG_RAW_HANDOFF_ACCURACY_METERS = 5.0;
    static final double RELAXED_RAW_HANDOFF_ACCURACY_METERS = 10.0;
    static final double EXCELLENT_ACCURACY_METERS = 4.0;
    static final double MAX_EARLY_HANDOFF_SPEED_MPS = 10.0;
    static final double MAX_EXCELLENT_ACCURACY_ENTRY_HEADING_ERROR_DEGREES = 35.0;
    static final double BACKWARD_PROGRESS_TOLERANCE_METERS = 8.0;
    static final long JUNCTION_HOLD_MS = 5_000L;
    static final long CONNECTED_RECOVERY_MIN_MS = 5_000L;
    static final long CONNECTED_RECOVERY_OUTLIER_PROTECTION_MS = 12_000L;
    static final long NO_EXIT_RELAXED_HANDOFF_MS = 10_000L;
    static final long DIRECTION_MIN_OBSERVATION_MS = 2_000L;
    static final double DIRECTION_MIN_MOVEMENT_METERS = 3.0;
    static final double DIRECTION_MAX_ERROR_DEGREES = 65.0;
    static final double DIRECTION_WIN_MARGIN_DEGREES = 25.0;
    static final long ENTRY_CONFIRMATION_WINDOW_MS = 5_000L;
    static final double ENTRY_MIN_CLOSING_METERS = 1.5;
    static final double ENTRY_STRONG_CLOSING_METERS = 3.0;
    static final double ENTRY_MIN_MOVEMENT_METERS = 2.0;
    static final double ENTRY_MAX_LOOSE_HEADING_ERROR_DEGREES = 55.0;
    static final double ENTRY_MAX_APPROACH_TO_ENTRANCE_ERROR_DEGREES = 75.0;
    static final double ENTRY_SHARP_REVERSAL_DEGREES = 135.0;
    static final long ENTRY_LOOSE_MIN_OBSERVATION_MS = 1_500L;

    enum State {
        NORMAL,
        STRUCTURE,
        JUNCTION_HOLD,
        RECOVERY,
        RAW_RECOVERY
    }

    static final class Point {
        final double lat;
        final double lon;

        Point(double lat, double lon) {
            this.lat = lat;
            this.lon = lon;
        }
    }

    static final class Segment {
        final String id;
        final String brunnel;
        final String subclass;
        final List<Point> points;

        Segment(String id, String brunnel, String subclass, List<Point> points) {
            this.id = id;
            this.brunnel = brunnel == null ? "" : brunnel;
            this.subclass = subclass == null ? "" : subclass;
            this.points = points;
        }

        boolean isStructure() {
            return "tunnel".equals(brunnel) || "bridge".equals(brunnel);
        }
    }

    static final class Context {
        final long updatedAtMs;
        final List<Segment> segments;

        Context(long updatedAtMs, List<Segment> segments) {
            this.updatedAtMs = updatedAtMs;
            this.segments = segments;
        }
    }

    static final class Result {
        final Point point;
        final boolean matched;
        final State state;
        final String structureType;

        Result(Point point, boolean matched, State state, String structureType) {
            this.point = point;
            this.matched = matched;
            this.state = state;
            this.structureType = structureType;
        }
    }

    private static final class Projection {
        final Point point;
        final double distanceMeters;
        final double alongMeters;

        Projection(Point point, double distanceMeters, double alongMeters) {
            this.point = point;
            this.distanceMeters = distanceMeters;
            this.alongMeters = alongMeters;
        }
    }

    private static final class SpeedSample {
        final long timeMs;
        final double speedMps;

        SpeedSample(long timeMs, double speedMps) {
            this.timeMs = timeMs;
            this.speedMps = speedMps;
        }
    }

    private static final class Observation {
        final Point point;
        final long timeMs;

        Observation(Point point, long timeMs) {
            this.point = point;
            this.timeMs = timeMs;
        }
    }

    private static final class Branch {
        final Segment segment;
        final boolean startsAtJunction;
        final List<Point> points;
        final double heading;

        Branch(Segment segment, boolean startsAtJunction) {
            this.segment = segment;
            this.startsAtJunction = startsAtJunction;
            this.points = orientedPoints(segment, startsAtJunction);
            this.heading = bearingDegrees(points.get(0), points.get(1));
        }
    }

    private static final class EntryEvidence {
        final double movementBearing;
        final double headingError;
        final double entranceApproachError;
        final boolean approachTrend;
        final boolean strongApproachTrend;
        final boolean recentSharpReversal;
        final long observationSpanMs;

        EntryEvidence(
                double movementBearing,
                double headingError,
                double entranceApproachError,
                boolean approachTrend,
                boolean strongApproachTrend,
                boolean recentSharpReversal,
                long observationSpanMs
        ) {
            this.movementBearing = movementBearing;
            this.headingError = headingError;
            this.entranceApproachError = entranceApproachError;
            this.approachTrend = approachTrend;
            this.strongApproachTrend = strongApproachTrend;
            this.recentSharpReversal = recentSharpReversal;
            this.observationSpanMs = observationSpanMs;
        }
    }

    private State state = State.NORMAL;
    private Context context;
    private Point lastRawPoint;
    private long lastRawTimeMs;
    private final Deque<SpeedSample> recentSpeeds = new ArrayDeque<>();
    private final Deque<Observation> entryObservations = new ArrayDeque<>();
    private final List<Observation> directionObservations = new ArrayList<>();
    private List<Point> activeRoute = Collections.emptyList();
    private List<Point> structureRoute = Collections.emptyList();
    private List<Branch> exitBranches = Collections.emptyList();
    private double activeRouteLengthMeters;
    private double progressMeters;
    private long lastMatchedTimeMs;
    private long junctionHoldStartedAtMs;
    private long connectedRecoveryStartedAtMs;
    private long noExitRecoveryStartedAtMs;
    private Point lastMatchedOutputPoint;
    private long lastMatchedOutputTimeMs;
    private double heldDistanceMeters;
    private String structureType = "";
    private String activeStructureId = "";

    void reset() {
        state = State.NORMAL;
        context = null;
        lastRawPoint = null;
        lastRawTimeMs = 0L;
        recentSpeeds.clear();
        entryObservations.clear();
        clearActiveRoute();
    }

    void setContext(Context context) {
        this.context = context;
    }

    State getState() {
        return state;
    }

    Result process(
            Point rawPoint,
            float accuracyMeters,
            long timeMs,
            boolean accuracyIsGood
    ) {
        updateRecentSpeed(rawPoint, timeMs, accuracyIsGood);
        updateEntryObservations(rawPoint, timeMs);

        if (state == State.RAW_RECOVERY) {
            Result result = new Result(rawPoint, false, state, "");
            if (rawHandoffIsStrong(accuracyMeters)) {
                state = State.NORMAL;
                result = new Result(rawPoint, false, state, "");
                clearActiveRoute();
            }
            lastRawPoint = rawPoint;
            lastRawTimeMs = timeMs;
            return result;
        }

        if (state == State.NORMAL) {
            tryEnterStructure(rawPoint, accuracyMeters, timeMs);
        }

        Result result;
        if (state == State.NORMAL) {
            result = new Result(rawPoint, false, state, "");
        } else if (state == State.JUNCTION_HOLD) {
            result = processJunctionHold(rawPoint, accuracyMeters, timeMs, accuracyIsGood);
        } else {
            advanceProgress(timeMs);
            Point predicted = pointAlong(activeRoute, progressMeters);

            if (state == State.STRUCTURE && progressMeters >= activeRouteLengthMeters) {
                result = reachStructureExit(rawPoint, accuracyMeters, timeMs, accuracyIsGood);
            } else if (state == State.RECOVERY) {
                result = processRecovery(
                        rawPoint,
                        accuracyMeters,
                        timeMs,
                        accuracyIsGood,
                        predicted
                );
            } else {
                result = new Result(predicted, true, state, structureType);
            }
        }

        if (result.matched) {
            lastMatchedOutputPoint = result.point;
            lastMatchedOutputTimeMs = timeMs;
        }
        lastRawPoint = rawPoint;
        lastRawTimeMs = timeMs;
        return result;
    }

    private Result reachStructureExit(
            Point rawPoint,
            float accuracyMeters,
            long timeMs,
            boolean accuracyIsGood
    ) {
        Point junction = structureRoute.get(structureRoute.size() - 1);
        exitBranches = findExitBranches(junction);

        if (exitBranches.size() == 1) {
            selectBranch(exitBranches.get(0), 0.0, timeMs);
            return processRecovery(
                    rawPoint,
                    accuracyMeters,
                    timeMs,
                    accuracyIsGood,
                    pointAlong(activeRoute, progressMeters)
            );
        }

        if (exitBranches.size() > 1) {
            state = State.JUNCTION_HOLD;
            junctionHoldStartedAtMs = timeMs;
            lastMatchedTimeMs = timeMs;
            heldDistanceMeters = 0.0;
            directionObservations.clear();
            directionObservations.add(new Observation(rawPoint, timeMs));
            return new Result(junction, true, state, structureType);
        }

        state = State.RECOVERY;
        connectedRecoveryStartedAtMs = 0L;
        noExitRecoveryStartedAtMs = timeMs;
        return processRecovery(rawPoint, accuracyMeters, timeMs, accuracyIsGood, junction);
    }

    private Result processJunctionHold(
            Point rawPoint,
            float accuracyMeters,
            long timeMs,
            boolean accuracyIsGood
    ) {
        Point junction = structureRoute.get(structureRoute.size() - 1);
        double elapsedSeconds = Math.max(0.0, (timeMs - lastMatchedTimeMs) / 1000.0);
        heldDistanceMeters += estimatedSpeedMps() * elapsedSeconds;
        lastMatchedTimeMs = timeMs;
        directionObservations.add(new Observation(rawPoint, timeMs));

        if (rawHandoffIsStrong(accuracyMeters)) {
            state = State.NORMAL;
            Result result = new Result(rawPoint, false, state, "");
            clearActiveRoute();
            return result;
        }

        Branch selected = selectBranchFromObservedDirection();
        if (selected != null) {
            selectBranch(selected, heldDistanceMeters, timeMs);
            return new Result(
                    pointAlong(activeRoute, progressMeters),
                    true,
                    state,
                    structureType
            );
        }

        if (timeMs - junctionHoldStartedAtMs < JUNCTION_HOLD_MS) {
            return new Result(junction, true, state, structureType);
        }

        state = State.RAW_RECOVERY;
        return new Result(rawPoint, false, state, "");
    }

    private Result processRecovery(
            Point rawPoint,
            float accuracyMeters,
            long timeMs,
            boolean accuracyIsGood,
            Point predicted
    ) {
        if (rawHandoffIsStrong(accuracyMeters)
                && connectedRecoveryHandoffIsAllowed(rawPoint, timeMs)) {
            state = State.NORMAL;
            Result result = new Result(rawPoint, false, state, "");
            clearActiveRoute();
            return result;
        }

        if (noExitRelaxedHandoffIsAllowed(accuracyMeters, timeMs)) {
            state = State.NORMAL;
            Result result = new Result(rawPoint, false, state, "");
            clearActiveRoute();
            return result;
        }

        return new Result(predicted, true, state, structureType);
    }

    private void tryEnterStructure(Point rawPoint, float accuracyMeters, long timeMs) {
        if (context == null
                || lastRawPoint == null
                || context.segments.isEmpty()) {
            return;
        }

        Segment best = null;
        boolean enterAtStart = true;
        double bestDistance = Double.MAX_VALUE;

        for (Segment segment : context.segments) {
            if (!segment.isStructure() || segment.points.size() < 2) continue;

            double startDistance = distanceMeters(rawPoint, segment.points.get(0));
            double endDistance = distanceMeters(
                    rawPoint,
                    segment.points.get(segment.points.size() - 1)
            );

            if (startDistance < bestDistance) {
                best = segment;
                enterAtStart = true;
                bestDistance = startDistance;
            }
            if (endDistance < bestDistance) {
                best = segment;
                enterAtStart = false;
                bestDistance = endDistance;
            }
        }

        if (best == null || bestDistance > ENTRY_RADIUS_METERS) return;

        Point entrance = enterAtStart
                ? best.points.get(0)
                : best.points.get(best.points.size() - 1);
        List<Point> orientedStructure = orientedPoints(best, enterAtStart);
        double structureBearing = bearingDegrees(
                orientedStructure.get(0),
                orientedStructure.get(1)
        );
        EntryEvidence entryEvidence = entryEvidence(
                rawPoint,
                entrance,
                structureBearing
        );
        if (entryEvidence == null || !entryEvidence.approachTrend) return;
        if (entryEvidence.headingError > 75.0) return;
        if (entryNeedsStrongerEvidence(
                accuracyMeters,
                bestDistance,
                entryEvidence
        )) {
            return;
        }

        structureRoute = orientedStructure;
        activeRoute = new ArrayList<>(structureRoute);
        activeRouteLengthMeters = polylineLength(activeRoute);
        progressMeters = Math.min(
                activeRouteLengthMeters,
                Math.max(0.0, project(activeRoute, rawPoint).alongMeters)
        );
        lastMatchedTimeMs = timeMs;
        structureType = best.brunnel;
        activeStructureId = best.id;
        state = State.STRUCTURE;
    }

    private List<Branch> findExitBranches(Point junction) {
        if (context == null) return Collections.emptyList();

        List<Branch> branches = new ArrayList<>();
        Set<String> seen = new HashSet<>();
        for (Segment segment : context.segments) {
            if (segment.points.size() < 2
                    || segment.id.equals(activeStructureId)
                    || segment.isStructure()) {
                continue;
            }

            double startDistance = distanceMeters(junction, segment.points.get(0));
            double endDistance = distanceMeters(
                    junction,
                    segment.points.get(segment.points.size() - 1)
            );

            if (startDistance <= CONNECTION_RADIUS_METERS && seen.add(segment.id + ":start")) {
                branches.add(new Branch(segment, true));
            }
            if (endDistance <= CONNECTION_RADIUS_METERS && seen.add(segment.id + ":end")) {
                branches.add(new Branch(segment, false));
            }
        }
        return branches;
    }

    private Branch selectBranchFromObservedDirection() {
        if (directionObservations.size() < 2) return null;

        Observation first = directionObservations.get(0);
        Observation last = directionObservations.get(directionObservations.size() - 1);
        if (last.timeMs - first.timeMs < DIRECTION_MIN_OBSERVATION_MS) return null;
        if (distanceMeters(first.point, last.point) < DIRECTION_MIN_MOVEMENT_METERS) return null;

        double observedBearing = bearingDegrees(first.point, last.point);
        Branch best = null;
        double bestError = Double.MAX_VALUE;
        double secondBestError = Double.MAX_VALUE;

        for (Branch branch : exitBranches) {
            double error = angleDifference(observedBearing, branch.heading);
            if (error < bestError) {
                secondBestError = bestError;
                bestError = error;
                best = branch;
            } else if (error < secondBestError) {
                secondBestError = error;
            }
        }

        if (best == null || bestError > DIRECTION_MAX_ERROR_DEGREES) return null;
        if (exitBranches.size() > 1
                && secondBestError - bestError < DIRECTION_WIN_MARGIN_DEGREES) {
            return null;
        }
        return best;
    }

    private void selectBranch(Branch branch, double initialDistanceMeters, long timeMs) {
        activeRoute = buildBranchContinuation(branch);
        activeRouteLengthMeters = polylineLength(activeRoute);
        progressMeters = Math.min(activeRouteLengthMeters, initialDistanceMeters);
        lastMatchedTimeMs = timeMs;
        state = State.RECOVERY;
        directionObservations.clear();
        exitBranches = Collections.emptyList();
        heldDistanceMeters = 0.0;
        connectedRecoveryStartedAtMs = timeMs;
        noExitRecoveryStartedAtMs = 0L;
    }

    private List<Point> buildBranchContinuation(Branch initialBranch) {
        List<Point> route = new ArrayList<>(initialBranch.points);
        Set<String> usedIds = new HashSet<>();
        usedIds.add(activeStructureId);
        usedIds.add(initialBranch.segment.id);

        for (int depth = 0; depth < 3; depth++) {
            Point routeEnd = route.get(route.size() - 1);
            double incomingBearing = bearingDegrees(route.get(route.size() - 2), routeEnd);
            Segment best = null;
            boolean startsAtEnd = true;
            double bestScore = Double.MAX_VALUE;

            for (Segment candidate : context.segments) {
                if (candidate.points.size() < 2
                        || candidate.isStructure()
                        || usedIds.contains(candidate.id)) {
                    continue;
                }

                double startDistance = distanceMeters(routeEnd, candidate.points.get(0));
                double endDistance = distanceMeters(
                        routeEnd,
                        candidate.points.get(candidate.points.size() - 1)
                );

                if (startDistance <= CONNECTION_RADIUS_METERS) {
                    double score = startDistance + angleDifference(
                            incomingBearing,
                            bearingDegrees(candidate.points.get(0), candidate.points.get(1))
                    ) / 20.0;
                    if (score < bestScore) {
                        best = candidate;
                        startsAtEnd = true;
                        bestScore = score;
                    }
                }
                if (endDistance <= CONNECTION_RADIUS_METERS) {
                    double score = endDistance + angleDifference(
                            incomingBearing,
                            bearingDegrees(
                                    candidate.points.get(candidate.points.size() - 1),
                                    candidate.points.get(candidate.points.size() - 2)
                            )
                    ) / 20.0;
                    if (score < bestScore) {
                        best = candidate;
                        startsAtEnd = false;
                        bestScore = score;
                    }
                }
            }

            if (best == null) break;
            List<Point> continuation = orientedPoints(best, startsAtEnd);
            route.addAll(continuation.subList(1, continuation.size()));
            usedIds.add(best.id);
        }
        return route;
    }

    private void updateRecentSpeed(Point rawPoint, long timeMs, boolean accuracyIsGood) {
        if (lastRawPoint != null && lastRawTimeMs > 0L && accuracyIsGood) {
            double elapsedSeconds = (timeMs - lastRawTimeMs) / 1000.0;
            if (elapsedSeconds > 0.0 && elapsedSeconds <= 5.0) {
                double speedMps = distanceMeters(lastRawPoint, rawPoint) / elapsedSeconds;
                if (speedMps >= 0.5 && speedMps <= 8.0) {
                    recentSpeeds.addLast(new SpeedSample(timeMs, speedMps));
                }
            }
        }

        while (!recentSpeeds.isEmpty()
                && timeMs - recentSpeeds.peekFirst().timeMs > 20_000L) {
            recentSpeeds.removeFirst();
        }
    }

    private void updateEntryObservations(Point rawPoint, long timeMs) {
        entryObservations.addLast(new Observation(rawPoint, timeMs));
        while (!entryObservations.isEmpty()
                && timeMs - entryObservations.peekFirst().timeMs > ENTRY_CONFIRMATION_WINDOW_MS) {
            entryObservations.removeFirst();
        }
    }

    private void advanceProgress(long timeMs) {
        if (lastMatchedTimeMs <= 0L) {
            lastMatchedTimeMs = timeMs;
            return;
        }

        double elapsedSeconds = Math.max(0.0, (timeMs - lastMatchedTimeMs) / 1000.0);
        progressMeters = Math.min(
                activeRouteLengthMeters,
                progressMeters + estimatedSpeedMps() * elapsedSeconds
        );
        lastMatchedTimeMs = timeMs;
    }

    private double estimatedSpeedMps() {
        if (recentSpeeds.isEmpty()) return 2.5;

        List<Double> speeds = new ArrayList<>();
        for (SpeedSample sample : recentSpeeds) {
            speeds.add(sample.speedMps);
        }
        speeds.sort(Comparator.naturalOrder());
        return speeds.get(speeds.size() / 2);
    }

    private void clearActiveRoute() {
        activeRoute = Collections.emptyList();
        structureRoute = Collections.emptyList();
        exitBranches = Collections.emptyList();
        directionObservations.clear();
        activeRouteLengthMeters = 0.0;
        progressMeters = 0.0;
        lastMatchedTimeMs = 0L;
        junctionHoldStartedAtMs = 0L;
        connectedRecoveryStartedAtMs = 0L;
        noExitRecoveryStartedAtMs = 0L;
        lastMatchedOutputPoint = null;
        lastMatchedOutputTimeMs = 0L;
        heldDistanceMeters = 0.0;
        structureType = "";
        activeStructureId = "";
    }

    private static boolean rawHandoffIsStrong(float accuracyMeters) {
        return accuracyMeters >= 0.0f
                && accuracyMeters <= STRONG_RAW_HANDOFF_ACCURACY_METERS;
    }

    private boolean noExitRelaxedHandoffIsAllowed(float accuracyMeters, long timeMs) {
        return noExitRecoveryStartedAtMs > 0L
                && timeMs - noExitRecoveryStartedAtMs >= NO_EXIT_RELAXED_HANDOFF_MS
                && accuracyMeters >= 0.0f
                && accuracyMeters <= RELAXED_RAW_HANDOFF_ACCURACY_METERS;
    }

    private boolean connectedRecoveryHandoffIsAllowed(Point rawPoint, long timeMs) {
        if (connectedRecoveryStartedAtMs <= 0L) return true;

        long recoveryAgeMs = timeMs - connectedRecoveryStartedAtMs;
        if (recoveryAgeMs < CONNECTED_RECOVERY_MIN_MS) return false;
        if (recoveryAgeMs >= CONNECTED_RECOVERY_OUTLIER_PROTECTION_MS) return true;

        return !earlyHandoffLooksLikeOutlier(rawPoint, timeMs);
    }

    private boolean earlyHandoffLooksLikeOutlier(Point rawPoint, long timeMs) {
        if (lastMatchedOutputPoint == null || lastMatchedOutputTimeMs <= 0L) return false;

        double elapsedSeconds = (timeMs - lastMatchedOutputTimeMs) / 1000.0;
        if (elapsedSeconds <= 0.0 || elapsedSeconds > 5.0) return false;

        double speedMps = distanceMeters(lastMatchedOutputPoint, rawPoint) / elapsedSeconds;
        if (speedMps > MAX_EARLY_HANDOFF_SPEED_MPS) return true;

        Projection rawProjection = project(activeRoute, rawPoint);
        return rawProjection.alongMeters < progressMeters - BACKWARD_PROGRESS_TOLERANCE_METERS;
    }

    private EntryEvidence entryEvidence(
            Point rawPoint,
            Point entrance,
            double structureBearing
    ) {
        if (entryObservations.size() < 2) return null;

        Observation first = null;
        Observation previous = null;
        Double previousHeading = null;
        double firstDistance = Double.MAX_VALUE;
        int closingSteps = 0;
        boolean recentSharpReversal = false;

        for (Observation observation : entryObservations) {
            double distanceFromCurrent = distanceMeters(observation.point, rawPoint);
            if (first == null && distanceFromCurrent >= ENTRY_MIN_MOVEMENT_METERS) {
                first = observation;
                firstDistance = distanceMeters(observation.point, entrance);
            }

            if (previous != null) {
                double priorDistance = distanceMeters(previous.point, entrance);
                double currentDistance = distanceMeters(observation.point, entrance);
                if (priorDistance > currentDistance + 0.75) {
                    closingSteps++;
                }

                double heading = bearingDegrees(previous.point, observation.point);
                if (previousHeading != null
                        && angleDifference(previousHeading, heading)
                        >= ENTRY_SHARP_REVERSAL_DEGREES) {
                    recentSharpReversal = true;
                }
                previousHeading = heading;
            }
            previous = observation;
        }

        if (first == null) return null;

        double currentDistance = distanceMeters(rawPoint, entrance);
        double closingMeters = firstDistance - currentDistance;
        boolean approachTrend = closingMeters >= ENTRY_MIN_CLOSING_METERS
                || closingSteps >= 1;
        boolean strongApproachTrend = closingMeters >= ENTRY_STRONG_CLOSING_METERS
                || closingSteps >= 2;
        double movementBearing = bearingDegrees(first.point, rawPoint);
        double entranceBearing = bearingDegrees(rawPoint, entrance);

        return new EntryEvidence(
                movementBearing,
                angleDifference(movementBearing, structureBearing),
                angleDifference(movementBearing, entranceBearing),
                approachTrend,
                strongApproachTrend,
                recentSharpReversal,
                previous == null ? 0L : previous.timeMs - first.timeMs
        );
    }

    private static boolean entryNeedsStrongerEvidence(
            float accuracyMeters,
            double entryDistanceMeters,
            EntryEvidence evidence
    ) {
        boolean excellentAccuracy = accuracyMeters >= 0.0f
                && accuracyMeters <= EXCELLENT_ACCURACY_METERS;
        if (excellentAccuracy) {
            return evidence.headingError > MAX_EXCELLENT_ACCURACY_ENTRY_HEADING_ERROR_DEGREES;
        }

        if (entryDistanceMeters <= STRICT_ENTRY_RADIUS_METERS) return false;

        if (evidence.observationSpanMs < ENTRY_LOOSE_MIN_OBSERVATION_MS) return true;
        if (!evidence.strongApproachTrend) return true;
        if (evidence.headingError > ENTRY_MAX_LOOSE_HEADING_ERROR_DEGREES) return true;
        if (evidence.entranceApproachError > ENTRY_MAX_APPROACH_TO_ENTRANCE_ERROR_DEGREES) {
            return true;
        }
        return evidence.recentSharpReversal && !evidence.strongApproachTrend;
    }

    private static List<Point> orientedPoints(Segment segment, boolean startToEnd) {
        List<Point> points = new ArrayList<>(segment.points);
        if (!startToEnd) Collections.reverse(points);
        return points;
    }

    private static Projection project(List<Point> polyline, Point target) {
        if (polyline.isEmpty()) {
            return new Projection(target, Double.MAX_VALUE, 0.0);
        }

        Point bestPoint = polyline.get(0);
        double bestDistance = distanceMeters(bestPoint, target);
        double bestAlong = 0.0;
        double traversed = 0.0;

        for (int i = 1; i < polyline.size(); i++) {
            Point a = polyline.get(i - 1);
            Point b = polyline.get(i);
            double segmentLength = distanceMeters(a, b);
            if (segmentLength <= 0.0) continue;

            double refLatRadians = Math.toRadians((a.lat + b.lat + target.lat) / 3.0);
            double ax = Math.toRadians(a.lon) * Math.cos(refLatRadians);
            double ay = Math.toRadians(a.lat);
            double bx = Math.toRadians(b.lon) * Math.cos(refLatRadians);
            double by = Math.toRadians(b.lat);
            double tx = Math.toRadians(target.lon) * Math.cos(refLatRadians);
            double ty = Math.toRadians(target.lat);
            double dx = bx - ax;
            double dy = by - ay;
            double denominator = dx * dx + dy * dy;
            double t = denominator == 0.0
                    ? 0.0
                    : ((tx - ax) * dx + (ty - ay) * dy) / denominator;
            t = Math.max(0.0, Math.min(1.0, t));
            Point projected = new Point(
                    a.lat + (b.lat - a.lat) * t,
                    a.lon + (b.lon - a.lon) * t
            );
            double distance = distanceMeters(projected, target);
            if (distance < bestDistance) {
                bestDistance = distance;
                bestPoint = projected;
                bestAlong = traversed + segmentLength * t;
            }
            traversed += segmentLength;
        }

        return new Projection(bestPoint, bestDistance, bestAlong);
    }

    private static Point pointAlong(List<Point> polyline, double distanceMeters) {
        if (polyline.isEmpty()) return new Point(0.0, 0.0);
        if (distanceMeters <= 0.0) return polyline.get(0);

        double traversed = 0.0;
        for (int i = 1; i < polyline.size(); i++) {
            Point a = polyline.get(i - 1);
            Point b = polyline.get(i);
            double segmentLength = distanceMeters(a, b);
            if (traversed + segmentLength >= distanceMeters) {
                double fraction = segmentLength <= 0.0
                        ? 0.0
                        : (distanceMeters - traversed) / segmentLength;
                return new Point(
                        a.lat + (b.lat - a.lat) * fraction,
                        a.lon + (b.lon - a.lon) * fraction
                );
            }
            traversed += segmentLength;
        }
        return polyline.get(polyline.size() - 1);
    }

    private static double polylineLength(List<Point> points) {
        double total = 0.0;
        for (int i = 1; i < points.size(); i++) {
            total += distanceMeters(points.get(i - 1), points.get(i));
        }
        return total;
    }

    static double distanceMeters(Point a, Point b) {
        double earthRadius = 6_371_000.0;
        double lat1 = Math.toRadians(a.lat);
        double lat2 = Math.toRadians(b.lat);
        double deltaLat = lat2 - lat1;
        double deltaLon = Math.toRadians(b.lon - a.lon);
        double sinLat = Math.sin(deltaLat / 2.0);
        double sinLon = Math.sin(deltaLon / 2.0);
        double h = sinLat * sinLat
                + Math.cos(lat1) * Math.cos(lat2) * sinLon * sinLon;
        return 2.0 * earthRadius * Math.asin(Math.min(1.0, Math.sqrt(h)));
    }

    private static double bearingDegrees(Point from, Point to) {
        double lat1 = Math.toRadians(from.lat);
        double lat2 = Math.toRadians(to.lat);
        double deltaLon = Math.toRadians(to.lon - from.lon);
        double y = Math.sin(deltaLon) * Math.cos(lat2);
        double x = Math.cos(lat1) * Math.sin(lat2)
                - Math.sin(lat1) * Math.cos(lat2) * Math.cos(deltaLon);
        return (Math.toDegrees(Math.atan2(y, x)) + 360.0) % 360.0;
    }

    private static double angleDifference(double a, double b) {
        double difference = Math.abs(a - b) % 360.0;
        return difference > 180.0 ? 360.0 - difference : difference;
    }
