package com.kdidso.livestride;

import android.content.Intent;
import android.os.Build;

import com.getcapacitor.JSObject;
import com.getcapacitor.Plugin;
import com.getcapacitor.PluginCall;
import com.getcapacitor.PluginMethod;
import com.getcapacitor.annotation.CapacitorPlugin;

@CapacitorPlugin(name = "LiveStrideNativeTracker")
public class LiveStrideNativeTrackerPlugin extends Plugin {

    @PluginMethod
    public void ping(PluginCall call) {
        JSObject result = new JSObject();
        result.put("ok", true);
        result.put("message", "LiveStrideNativeTracker native plugin is loaded.");
        call.resolve(result);
    }

    @PluginMethod
    public void insertTestPoint(PluginCall call) {
        String supabaseUrl = call.getString("supabaseUrl");
        String supabaseAnonKey = call.getString("supabaseAnonKey");

        double lat = call.getDouble("lat", 37.1);
        double lon = call.getDouble("lon", -113.6);

        if (supabaseUrl == null || supabaseAnonKey == null) {
            call.reject("Missing Supabase URL or anon key.");
            return;
        }

        new Thread(() -> {
            try {
                org.json.JSONObject body = new org.json.JSONObject();
                body.put("runner_name", "Native Test");
                body.put("source", "native");
                body.put("native_session_id", "manual_test");
                body.put("sequence_num", 1);
                body.put("lat", lat);
                body.put("lon", lon);
                body.put("accuracy", 5.0);
                body.put("recorded_at", java.time.Instant.now().toString());
                body.put("app_state", "manual_plugin_test");
                body.put("note", "Inserted from native Android plugin");

                java.net.URL url = new java.net.URL(
                        supabaseUrl + "/rest/v1/run_points_native_test"
                );

                java.net.HttpURLConnection conn =
                        (java.net.HttpURLConnection) url.openConnection();

                conn.setRequestMethod("POST");
                conn.setRequestProperty("apikey", supabaseAnonKey);
                conn.setRequestProperty("Authorization", "Bearer " + supabaseAnonKey);
                conn.setRequestProperty("Content-Type", "application/json");
                conn.setRequestProperty("Accept", "application/json");
                conn.setRequestProperty("Accept-Profile", "public");
                conn.setRequestProperty("Content-Profile", "public");
                conn.setRequestProperty("Prefer", "return=minimal");
                conn.setDoOutput(true);

                try (java.io.OutputStream os = conn.getOutputStream()) {
                    os.write(body.toString().getBytes("UTF-8"));
                }

                int code = conn.getResponseCode();

                if (code >= 200 && code < 300) {
                    JSObject result = new JSObject();
                    result.put("ok", true);
                    result.put("status", code);
                    result.put("message", "Native test point inserted.");
                    call.resolve(result);
                } else {
                    call.reject("Supabase insert failed. HTTP status: " + code);
                }

                conn.disconnect();

            } catch (Exception e) {
                call.reject("Native insert exception: " + e.getMessage(), e);
            }
        }).start();
    }

    @PluginMethod
    public void startNativeTestRun(PluginCall call) {
        String supabaseUrl = call.getString("supabaseUrl");
        String supabaseAnonKey = call.getString("supabaseAnonKey");
        String runId = call.getString("runId");
        String runnerName = call.getString("runnerName", "Native Test Runner");
        String nativeSessionId = call.getString(
                "nativeSessionId",
                "native_" + System.currentTimeMillis()
        );

        if (supabaseUrl == null || supabaseAnonKey == null) {
            call.reject("Missing Supabase URL or anon key.");
            return;
        }

        Intent intent = new Intent(getContext(), LiveStrideNativeTrackerService.class);
        intent.setAction(LiveStrideNativeTrackerService.ACTION_START);
        intent.putExtra("supabaseUrl", supabaseUrl);
        intent.putExtra("supabaseAnonKey", supabaseAnonKey);
        intent.putExtra("runId", runId);
        intent.putExtra("runnerName", runnerName);
        intent.putExtra("nativeSessionId", nativeSessionId);

        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
            getContext().startForegroundService(intent);
        } else {
            getContext().startService(intent);
        }

        JSObject result = new JSObject();
        result.put("ok", true);
        result.put("message", "Native GPS test service started.");
        result.put("nativeSessionId", nativeSessionId);
        call.resolve(result);
    }

    @PluginMethod
    public void stopNativeTestRun(PluginCall call) {
        Intent intent = new Intent(getContext(), LiveStrideNativeTrackerService.class);
        intent.setAction(LiveStrideNativeTrackerService.ACTION_STOP);

        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
            getContext().startForegroundService(intent);
        } else {
            getContext().startService(intent);
        }

        JSObject result = new JSObject();
        result.put("ok", true);
        result.put("message", "Native GPS test service stop requested.");
        call.resolve(result);
    }

    @PluginMethod
    public void getNativeRunSnapshot(PluginCall call) {
        String runId = call.getString("runId");
        int afterSequenceNum = call.getInt("afterSequenceNum", -1);
        boolean includeFullPath = call.getBoolean("includeFullPath", false);

        if (runId == null || runId.trim().isEmpty()) {
            call.reject("Missing runId.");
            return;
        }

        new Thread(() -> {
            try {
                LiveStridePointStore pointStore = new LiveStridePointStore(getContext());
                org.json.JSONObject snapshot = pointStore.getRunDisplaySnapshot(
                        runId,
                        afterSequenceNum,
                        includeFullPath
                );

                JSObject result = new JSObject();
                result.put("ok", snapshot != null);
                result.put("snapshotJson", snapshot == null ? "{}" : snapshot.toString());
                call.resolve(result);
            } catch (Exception e) {
                call.reject("Native run snapshot failed: " + e.getMessage(), e);
            }
        }).start();
    }
}
