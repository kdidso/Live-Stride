    @Override
    public void onCreate() {
        super.onCreate();
        if (isLightPhoneIII()) {
            platformLocationManager = (LocationManager) getSystemService(LOCATION_SERVICE);
        } else {
            fusedLocationClient = LocationServices.getFusedLocationProviderClient(this);
        }
        pointStore = new LiveStridePointStore(this);
        mapContextProvider = new LiveStrideMapContextProvider(this);
        createNotificationChannel();
    }

    @Override
    public int onStartCommand(Intent intent, int flags, int startId) {
        if (intent == null || intent.getAction() == null) {
            Log.w("LiveStrideNative", "onStartCommand received null intent or null action");
            return START_NOT_STICKY;
        }

        Log.i(
                "LiveStrideNative",
                "onStartCommand action=" + intent.getAction()
                        + " hasExtras=" + (intent.getExtras() != null)
                        + " runId=" + intent.getStringExtra("runId")
                        + " keyLength=" + (
                        intent.getStringExtra("supabaseAnonKey") == null
                                ? 0
                                : intent.getStringExtra("supabaseAnonKey").length()
                )
        );

        if (ACTION_START.equals(intent.getAction())) {
            supabaseUrl = intent.getStringExtra("supabaseUrl");
            supabaseAnonKey = intent.getStringExtra("supabaseAnonKey");
            runId = intent.getStringExtra("runId");
            runnerName = intent.getStringExtra("runnerName");
            nativeSessionId = intent.getStringExtra("nativeSessionId");
            firebaseDatabaseUrl = intent.getStringExtra("firebaseDatabaseUrl");
            firebaseRunnerPath = intent.getStringExtra("firebaseRunnerPath");
            localOnly = intent.getBooleanExtra("localOnly", false);
            if (!localOnly) {
                LiveStridePendingPointUploadJobService.saveSupabaseConfig(
                        this,
                        supabaseUrl,
                        supabaseAnonKey
                );
            }

            if (runnerName == null || runnerName.trim().isEmpty()) {
                runnerName = localOnly ? "Private Run" : "Native Test Runner";
            }

            if (nativeSessionId == null || nativeSessionId.trim().isEmpty()) {
                nativeSessionId = (localOnly ? "private_" : "native_") + System.currentTimeMillis();
            }  

            Log.i("LiveStrideNative", "SQLite native point storage active. Pending before run=" + pointStore.countPending());
            pointStore.exportDebugCopy();
            LiveStridePendingPointUploadJobService.schedule(this);
            new Thread(() -> LiveStridePendingPointUploadJobService.flushPendingPoints(this)).start();

            sequenceNum = 0;
            previousLocation = null;
            totalDistanceMeters = 0.0;
            runStartTimeMs = 0L;
            previousPointTimeMs = 0L;
            loadedMapContextUpdatedAtMs = 0L;
            serverMapContextActive = false;
            lastCompletedMile = 0;
            lastMileSplitTimeMs = 0L;
            latestMileSplitText = "Mile split: --";
            bridgeTunnelMatcher.reset();
            saveMatchingState(BridgeTunnelMatcher.State.NORMAL);
            saveNativeRunConfig();
            saveNativeTrackerState(true, 0L, 0L);
            
            startForeground(NOTIFICATION_ID, buildRunNotification(
                    System.currentTimeMillis(),
                    0.0,
                    0.0,
                    latestMileSplitText,
                    false
            ));
            startLocationUpdates();

            return START_STICKY;
        }

        if (ACTION_STOP.equals(intent.getAction())) {
            restoreNativeRunConfig(intent);
            localOnly = intent.getBooleanExtra("localOnly", localOnly);
            stopLocationUpdates();
            saveMatchingState(BridgeTunnelMatcher.State.NORMAL);
            if (!localOnly) {
                queueNativeFinishEvent();
            }
            long stoppedAtMs = System.currentTimeMillis();
            long finalTimeMs = runStartTimeMs > 0L ? stoppedAtMs - runStartTimeMs : 0L;
            saveNativeTrackerState(false, stoppedAtMs, finalTimeMs);
            LiveStridePendingPointUploadJobService.schedule(this);
            new Thread(() -> {
                LiveStridePendingPointUploadJobService.flushPendingFinishEvents(this);
                LiveStridePendingPointUploadJobService.flushPendingPoints(this);
            }).start();
            stopForeground(true);
            stopSelf();
            return START_NOT_STICKY;
        }

        return START_NOT_STICKY;
    }

    private void startLocationUpdates() {
        if (isTracking || locationStartRequested) return;

        if (!localOnly && (supabaseUrl == null || supabaseAnonKey == null)) {
            stopSelf();
            return;
        }

        if (ActivityCompat.checkSelfPermission(this, Manifest.permission.ACCESS_FINE_LOCATION)
                != PackageManager.PERMISSION_GRANTED
                && ActivityCompat.checkSelfPermission(this, Manifest.permission.ACCESS_COARSE_LOCATION)
                != PackageManager.PERMISSION_GRANTED) {
            stopSelf();
            return;
        }

        if (isLightPhoneIII()) {
            startLightPhoneLocationUpdates();
            return;
        }

        LocationRequest locationRequest = new LocationRequest.Builder(
                Priority.PRIORITY_HIGH_ACCURACY,
                1000L
        )
                .setMinUpdateIntervalMillis(1000L)
                .setMinUpdateDistanceMeters(1.0f)
                .build();

        locationCallback = new LocationCallback() {
            @Override
            public void onLocationResult(LocationResult locationResult) {
                if (locationResult == null) return;

                for (Location location : locationResult.getLocations()) {
                    if (location == null) continue;
                    uploadNativePoint(location);
                }
            }
        };

        locationStartRequested = true;
        fusedLocationClient.requestLocationUpdates(
                locationRequest,
                locationCallback,
                Looper.getMainLooper()
        ).addOnSuccessListener(ignored -> {
            if (!locationStartRequested) return;
            isTracking = true;
            locationStartRequested = false;
            Log.i("LiveStrideNative", "Google fused location updates started");
        }).addOnFailureListener(error -> {
            isTracking = false;
            locationStartRequested = false;
            Log.e("LiveStrideNative", "Google fused location request failed", error);
        });
    }

    private void startLightPhoneLocationUpdates() {
        if (platformLocationManager == null) {
            Log.e("LiveStrideNative", "Light Phone location service is unavailable");
            return;
        }

        String provider = null;
        if (platformLocationManager.getProvider(LocationManager.GPS_PROVIDER) != null) {
            provider = LocationManager.GPS_PROVIDER;
        } else if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.S
                && platformLocationManager.getProvider(LocationManager.FUSED_PROVIDER) != null) {
            provider = LocationManager.FUSED_PROVIDER;
        } else if (platformLocationManager.getProvider(LocationManager.NETWORK_PROVIDER) != null) {
            provider = LocationManager.NETWORK_PROVIDER;
        }

        if (provider == null) {
            Log.e("LiveStrideNative", "Light Phone has no usable Android location provider");
            return;
        }

        platformLocationListener = new LocationListener() {
            @Override
            public void onLocationChanged(Location location) {
                if (location != null) uploadNativePoint(location);
            }

            @Override
            public void onProviderEnabled(String providerName) {
                Log.i("LiveStrideNative", "Light Phone location provider enabled: " + providerName);
            }

            @Override
            public void onProviderDisabled(String providerName) {
                Log.w("LiveStrideNative", "Light Phone location provider disabled: " + providerName);
            }

            @Override
            @SuppressWarnings("deprecation")
            public void onStatusChanged(String providerName, int status, Bundle extras) {
                // Retained for compatibility with older Android versions supported by LiveStride.
            }
        };

        try {
            locationStartRequested = true;
            platformLocationManager.requestLocationUpdates(
                    provider,
                    1000L,
                    1.0f,
                    platformLocationListener,
                    Looper.getMainLooper()
            );
            usingLightPhoneLocationManager = true;
            isTracking = true;
            locationStartRequested = false;
            Log.i(
                    "LiveStrideNative",
                    "Light Phone Android location updates started with provider=" + provider
            );
        } catch (SecurityException | IllegalArgumentException error) {
            usingLightPhoneLocationManager = false;
            isTracking = false;
            locationStartRequested = false;
            platformLocationListener = null;
            Log.e("LiveStrideNative", "Light Phone location request failed", error);
        }
    }

    private static boolean isLightPhoneIII() {
        return "Light".equalsIgnoreCase(Build.MANUFACTURER)
                && ("TLP301".equalsIgnoreCase(Build.MODEL)
                || "LightPhoneIII".equalsIgnoreCase(Build.DEVICE));
    }

    private void stopLocationUpdates() {
        locationStartRequested = false;
        if (usingLightPhoneLocationManager
                && platformLocationListener != null
                && platformLocationManager != null) {
            try {
                platformLocationManager.removeUpdates(platformLocationListener);
            } catch (SecurityException error) {
                Log.w("LiveStrideNative", "Could not remove Light Phone location updates", error);
            }
        }
        if (locationCallback != null && fusedLocationClient != null) {
            fusedLocationClient.removeLocationUpdates(locationCallback);
        }

        platformLocationListener = null;
        usingLightPhoneLocationManager = false;
        locationCallback = null;
        isTracking = false;
    }

private void uploadNativePoint(Location rawLocation) {
    final float accuracy = rawLocation.hasAccuracy() ? rawLocation.getAccuracy() : -1f;
    final long pointTimeMs = rawLocation.getTime() > 0
            ? rawLocation.getTime()
            : System.currentTimeMillis();
    final boolean accuracyIsGood = accuracy < 0
            || accuracy <= MAX_ACCEPTED_ACCURACY_METERS;

    if (!serverMapContextActive) {
        refreshMapMatchingContext();
    }
    mapContextProvider.requestForLocation(
            rawLocation.getLatitude(),
            rawLocation.getLongitude()
    );
    BridgeTunnelMatcher.Context serverContext =
            mapContextProvider.consumePendingContext();
    if (serverContext != null) {
        bridgeTunnelMatcher.setContext(serverContext);
        serverMapContextActive = true;
        Log.i(
                "LiveStrideNative",
                "Activated server map context with "
                        + serverContext.segments.size() + " segments"
        );
    }
    final boolean presumedDriverMode = isPresumedDriver(pointTimeMs);
    BridgeTunnelMatcher.Result matchResult = bridgeTunnelMatcher.process(
            new BridgeTunnelMatcher.Point(
                    rawLocation.getLatitude(),
                    rawLocation.getLongitude()
            ),
            accuracy,
            pointTimeMs,
            accuracyIsGood,
            presumedDriverMode
    );
    saveMatchingState(matchResult.state);

    if (!matchResult.matched && matcherIsHoldingForCleanerGps(matchResult.state)) {
        Log.i(
                "LiveStrideNative",
                "Holding raw GPS during map-matching recovery: "
                        + accuracy + " m state=" + matchResult.state
        );
        return;
    }

    if (!matchResult.matched && !accuracyIsGood) {
        Log.i("LiveStrideNative", "Ignored inaccurate GPS fix: " + accuracy + " m");
        return;
    }

    Location location = new Location(rawLocation);
    if (matchResult.matched) {
        location.setLatitude(matchResult.point.lat);
        location.setLongitude(matchResult.point.lon);
        Log.i(
                "LiveStrideNative",
                "Map-matched " + matchResult.structureType
                        + " point state=" + matchResult.state
                        + " rawAccuracy=" + accuracy
        );
    }

    final double lat = location.getLatitude();
    final double lon = location.getLongitude();
    final boolean wasMapMatched = matchResult.matched;
    final String matchingState = matchResult.state.name().toLowerCase(java.util.Locale.US);
    final String matchedStructureType = matchResult.structureType;

    double segmentMeters = 0.0;
    double instPaceMinPerMile = 0.0;
    double avgPaceMinPerMile = 0.0;
    boolean alertNotificationForMileSplit = false;

    if (runStartTimeMs == 0L) {
        runStartTimeMs = pointTimeMs;
        saveNativeTrackerState(true, 0L, 0L);
    }

    if (previousLocation != null && previousPointTimeMs > 0L) {
        segmentMeters = previousLocation.distanceTo(location);

        double elapsedSeconds = (pointTimeMs - previousPointTimeMs) / 1000.0;
        boolean resumeAfterLongGap = elapsedSeconds > RESUME_ANCHOR_GAP_SECONDS
                && segmentMeters > RESUME_ANCHOR_MIN_DISTANCE_METERS;

        if (resumeAfterLongGap) {
            segmentMeters = 0.0;
            System.out.println(
                    "LiveStride native anchored resume after long GPS gap: "
                            + elapsedSeconds + " s"
            );
        } else {
        double speedMps = elapsedSeconds > 0 ? segmentMeters / elapsedSeconds : 0.0;
        double speedMph = speedMps * 2.23694;

        if (segmentMeters < 1.0) {
            System.out.println("LiveStride native ignored tiny GPS movement: " + segmentMeters + " m");
            return;
        }

        if (elapsedSeconds <= 0.0) {
            System.out.println("LiveStride native ignored zero/negative time point: " + elapsedSeconds);
            return;
        }

        if (speedMph < 1) {
            System.out.println("LiveStride native ignored low-speed GPS drift: " + speedMph + " mph");
            return;
        }

        if (speedMph > 140.0) {
            System.out.println("LiveStride native ignored unrealistic GPS jump: " + speedMph + " mph");
            return;
        }

        double previousTotalDistanceMeters = totalDistanceMeters;
        totalDistanceMeters += segmentMeters;
        alertNotificationForMileSplit = updateMileSplit(
                previousTotalDistanceMeters,
                totalDistanceMeters,
                previousPointTimeMs,
                pointTimeMs
        );

        double segmentMiles = segmentMeters / METERS_PER_MILE;
        double elapsedMinutes = elapsedSeconds / 60.0;

        if (segmentMiles > 0) {
            instPaceMinPerMile = elapsedMinutes / segmentMiles;
        }
        }
    }

    double totalMiles = totalDistanceMeters / METERS_PER_MILE;
    double totalMinutes = (pointTimeMs - runStartTimeMs) / 60000.0;

    if (totalMiles > 0) {
        avgPaceMinPerMile = totalMinutes / totalMiles;
    }

    previousLocation = new Location(location);
    previousPointTimeMs = pointTimeMs;

    final int thisSequence = sequenceNum++;
    final double finalDistanceMeters = totalDistanceMeters;
    final double finalInstPaceMinPerMile = instPaceMinPerMile;
    final double finalAvgPaceMinPerMile = avgPaceMinPerMile;
    updateRunNotification(
            pointTimeMs,
            finalDistanceMeters,
            finalAvgPaceMinPerMile,
            latestMileSplitText,
            alertNotificationForMileSplit
    );
    java.text.SimpleDateFormat recordedAtFormatter =
        new java.text.SimpleDateFormat(
            "yyyy-MM-dd'T'HH:mm:ss.SSS'Z'",
            java.util.Locale.US
        );
    recordedAtFormatter.setTimeZone(java.util.TimeZone.getTimeZone("UTC"));
    final String recordedAt = recordedAtFormatter.format(new java.util.Date(pointTimeMs));

    new Thread(() -> {
        try {
            Double elevationMeters =
                    localOnly ? null : LiveStrideServerClient.fetchElevationMeters(lat, lon);
            JSONObject body = new JSONObject();

            if (runId != null && !runId.trim().isEmpty()) {
                body.put("run_id", runId);
            }

            body.put("runner_name", runnerName);
            body.put("source", localOnly ? "native_private" : "native");
            body.put("native_session_id", nativeSessionId);
            body.put("sequence_num", thisSequence);
            body.put("lat", lat);
            body.put("lon", lon);
            body.put("accuracy", accuracy >= 0 ? accuracy : JSONObject.NULL);
            body.put("recorded_at", recordedAt);
            body.put("distance_m", finalDistanceMeters);
            body.put("inst_pace_min_per_mile", finalInstPaceMinPerMile);
            body.put("avg_pace_min_per_mile", finalAvgPaceMinPerMile);
            body.put("elevation_m", elevationMeters != null ? elevationMeters : JSONObject.NULL);
            body.put(
                    "app_state",
                    localOnly
                            ? (wasMapMatched
                            ? "private_native_map_matched"
                            : "private_native")
                            : wasMapMatched
                            ? "foreground_service_map_matched"
                            : "foreground_service"
            );
            body.put(
                    "note",
                    localOnly
                            ? (wasMapMatched
                            ? "Private local run point map matched to connected "
                                    + matchedStructureType + " route during " + matchingState
                            : "Private local run point stored only on this phone")
                            : wasMapMatched
                            ? "Map matched to connected " + matchedStructureType
                                    + " route during " + matchingState
                            : "Inserted from native Android GPS service with SQLite-backed local storage"
            );

            // Phase SQLite-1:
            // Save every native point locally first. This makes the phone-side record durable
            // before any Supabase network attempt is made.
            long localId = pointStore.insertNativePoint(body);
            if (localOnly) {
                pointStore.markLocalOnly(localId);
                Double privateElevationMeters =
                        LiveStrideServerClient.fetchElevationMeters(lat, lon);
                if (privateElevationMeters != null) {
                    pointStore.updateElevation(localId, privateElevationMeters);
                }
                Log.i("LiveStrideNative", "Private local point saved seq=" + thisSequence
                        + " local_id=" + localId);
                return;
            }
            pointStore.markUploadInProgress(localId);

            LiveStridePendingPointUploadJobService.UploadResult uploadResult =
                    LiveStridePendingPointUploadJobService.uploadNativePointBody(
                            supabaseUrl,
                            supabaseAnonKey,
                            body
                    );

            if (uploadResult.success) {
                pointStore.markUploaded(localId);
                Log.i("LiveStrideNative", "SQLite point uploaded seq=" + thisSequence + " local_id=" + localId);
                kickPendingFlushIfNeeded();
            } else {
                pointStore.markUploadFailed(localId, uploadResult.error);
                LiveStridePendingPointUploadJobService.schedule(this);
                Log.w("LiveStrideNative", "SQLite point left pending seq=" + thisSequence
                        + " local_id=" + localId
                        + " pendingForRun=" + pointStore.countPendingForRun(runId));
            }

        } catch (Exception e) {
            System.out.println("LiveStride native point SQLite/upload exception: " + e.getMessage());
        }
    }).start();
}

private void saveMatchingState(BridgeTunnelMatcher.State state) {
    if (state == lastSavedMatchingState) return;
    lastSavedMatchingState = state;
    getSharedPreferences(STATE_PREFS_NAME, MODE_PRIVATE)
            .edit()
            .putString(STATE_MATCHING_STATE, state.name().toLowerCase(java.util.Locale.US))
            .apply();
}

private boolean matcherIsHoldingForCleanerGps(BridgeTunnelMatcher.State state) {
    return state == BridgeTunnelMatcher.State.RAW_RECOVERY;
}

private void refreshMapMatchingContext() {
    SharedPreferences prefs = getSharedPreferences(STATE_PREFS_NAME, MODE_PRIVATE);
    long updatedAtMs = prefs.getLong(STATE_MAP_CONTEXT_UPDATED_AT_MS, 0L);
    if (updatedAtMs <= loadedMapContextUpdatedAtMs) return;

    loadedMapContextUpdatedAtMs = updatedAtMs;
    String contextRunId = prefs.getString(STATE_MAP_CONTEXT_RUN_ID, null);
    if (runId == null || contextRunId == null || !runId.equals(contextRunId)) return;

    String contextJson = prefs.getString(STATE_MAP_CONTEXT_JSON, null);
    if (contextJson == null || contextJson.trim().isEmpty()) return;

    try {
        JSONObject root = new JSONObject(contextJson);
        JSONArray jsonSegments = root.optJSONArray("segments");
        if (jsonSegments == null) return;

        List<BridgeTunnelMatcher.Segment> segments = new ArrayList<>();
        for (int i = 0; i < jsonSegments.length(); i++) {
            JSONObject jsonSegment = jsonSegments.optJSONObject(i);
            if (jsonSegment == null) continue;

            JSONArray jsonPoints = jsonSegment.optJSONArray("points");
            if (jsonPoints == null || jsonPoints.length() < 2) continue;

            List<BridgeTunnelMatcher.Point> points = new ArrayList<>();
            for (int j = 0; j < jsonPoints.length(); j++) {
                JSONArray coordinate = jsonPoints.optJSONArray(j);
                if (coordinate == null || coordinate.length() < 2) continue;
                points.add(new BridgeTunnelMatcher.Point(
                        coordinate.optDouble(1),
                        coordinate.optDouble(0)
                ));
            }

            if (points.size() < 2) continue;
            segments.add(new BridgeTunnelMatcher.Segment(
                    jsonSegment.optString("id", "segment_" + i),
                    jsonSegment.optString("brunnel", ""),
                    jsonSegment.optString("subclass", ""),
                    points
            ));
        }

        bridgeTunnelMatcher.setContext(
                new BridgeTunnelMatcher.Context(updatedAtMs, segments)
        );
        Log.i(
                "LiveStrideNative",
                "Loaded bridge/tunnel map context with " + segments.size() + " segments"
        );
    } catch (Exception e) {
        Log.w("LiveStrideNative", "Could not parse map matching context: " + e.getMessage());
    }
}

private void queueNativeFinishEvent() {
    if (runId == null || runId.trim().isEmpty()) {
        Log.w("LiveStrideNative", "Stop requested without runId; finish event not queued");
        return;
    }

    try {
        String finishPayloadJson = buildNativeFinishPayload();
        pointStore.savePendingFinishEvent(
                runId,
                supabaseUrl,
                supabaseAnonKey,
                firebaseDatabaseUrl,
                firebaseRunnerPath,
                finishPayloadJson
        );
    } catch (Exception e) {
        Log.w("LiveStrideNative", "Could not queue native finish event: " + e.getMessage());
    }
}

private String buildNativeFinishPayload() throws org.json.JSONException {
    JSONObject finishPayload = pointStore.getRunSummaryFromNativePoints(runId);
    if (finishPayload == null) {
        finishPayload = new JSONObject();
    }

    String finishedAt = isoNow();
    finishPayload.put("is_live", false);
    finishPayload.put("finished_at", finishedAt);
    finishPayload.put("finalization_status", "completed");
    finishPayload.put("finalization_started_at", finishedAt);
    finishPayload.put("finalization_completed_at", finishedAt);

    if (!finishPayload.has("final_time_ms") || finishPayload.isNull("final_time_ms")) {
        long elapsedMs = runStartTimeMs > 0L
                ? Math.max(0L, System.currentTimeMillis() - runStartTimeMs)
                : 0L;
        finishPayload.put("final_time_ms", elapsedMs);
    }

    if (!finishPayload.has("total_distance_m") || finishPayload.isNull("total_distance_m")) {
        finishPayload.put("total_distance_m", totalDistanceMeters);
    }

    return finishPayload.toString();
}

private void kickPendingFlushIfNeeded() {
    if (pointStore.countPendingForRun(runId) <= 0) {
        return;
    }

    long now = System.currentTimeMillis();
    if (now - lastPendingFlushKickMs < 3_000L) {
        return;
    }

    lastPendingFlushKickMs = now;
    new Thread(() -> {
        Log.i("LiveStrideNative", "Immediate pending native point flush kicked after reconnect");
        LiveStridePendingPointUploadJobService.flushPendingPoints(this);
    }).start();
}

private boolean isPresumedDriver(long pointTimeMs) {
    if (runStartTimeMs <= 0L || totalDistanceMeters <= 0.0) {
        return false;
    }

    double elapsedSeconds = (pointTimeMs - runStartTimeMs) / 1000.0;
    if (elapsedSeconds <= 0.0) {
        return false;
    }

    return totalDistanceMeters / elapsedSeconds > PRESUMED_DRIVER_AVG_SPEED_MPS;
}

private boolean updateMileSplit(
        double previousTotalMeters,
        double currentTotalMeters,
        long segmentStartTimeMs,
        long segmentEndTimeMs
) {
    if (runStartTimeMs <= 0L || segmentStartTimeMs <= 0L || segmentEndTimeMs <= segmentStartTimeMs) {
        return false;
    }

    int completedMile = (int) Math.floor(currentTotalMeters / METERS_PER_MILE);
    if (completedMile <= lastCompletedMile) {
        return false;
    }

    double segmentMeters = currentTotalMeters - previousTotalMeters;
    if (segmentMeters <= 0.0) {
        return false;
    }

    boolean completedNewMile = false;

    for (int mile = lastCompletedMile + 1; mile <= completedMile; mile++) {
        double targetMeters = mile * METERS_PER_MILE;
        double segmentFraction = (targetMeters - previousTotalMeters) / segmentMeters;
        segmentFraction = Math.max(0.0, Math.min(1.0, segmentFraction));
        long splitTimeMs = segmentStartTimeMs
                + Math.round((segmentEndTimeMs - segmentStartTimeMs) * segmentFraction);
        long previousSplitTimeMs = lastMileSplitTimeMs > 0L
                ? lastMileSplitTimeMs
                : runStartTimeMs;
        double splitMinutes = (splitTimeMs - previousSplitTimeMs) / 60000.0;

        if (splitMinutes > 0.0) {
            latestMileSplitText = "Mile " + mile + ": " + formatPace(splitMinutes);
            completedNewMile = true;
        }

        lastCompletedMile = mile;
        lastMileSplitTimeMs = splitTimeMs;
    }

    return completedNewMile;
}

private void updateRunNotification(
        long pointTimeMs,
        double distanceMeters,
        double avgPaceMinPerMile,
        String mileSplitText,
        boolean alertThisUpdate
) {
    NotificationManager manager = getSystemService(NotificationManager.class);
    if (manager == null) {
        return;
    }

    manager.notify(
            NOTIFICATION_ID,
            buildRunNotification(pointTimeMs, distanceMeters, avgPaceMinPerMile, mileSplitText, alertThisUpdate)
    );
}

    private Notification buildRunNotification(
            long nowMs,
            double distanceMeters,
            double avgPaceMinPerMile,
            String mileSplitText,
            boolean alertThisUpdate
    ) {
        Intent stopIntent = new Intent(this, LiveStrideNativeTrackerService.class);
        stopIntent.setAction(ACTION_STOP);
        putStringExtraIfPresent(stopIntent, "supabaseUrl", supabaseUrl);
        putStringExtraIfPresent(stopIntent, "supabaseAnonKey", supabaseAnonKey);
        putStringExtraIfPresent(stopIntent, "runId", runId);
        putStringExtraIfPresent(stopIntent, "runnerName", runnerName);
        putStringExtraIfPresent(stopIntent, "nativeSessionId", nativeSessionId);
        putStringExtraIfPresent(stopIntent, "firebaseDatabaseUrl", firebaseDatabaseUrl);
        putStringExtraIfPresent(stopIntent, "firebaseRunnerPath", firebaseRunnerPath);

        int flags = PendingIntent.FLAG_UPDATE_CURRENT;
        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.M) {
            flags |= PendingIntent.FLAG_IMMUTABLE;
        }

        PendingIntent stopPendingIntent = PendingIntent.getService(
                this,
                44224,
                stopIntent,
                flags
        );

        long elapsedMs = runStartTimeMs > 0L
                ? Math.max(0L, nowMs - runStartTimeMs)
                : 0L;
        double distanceMiles = distanceMeters / METERS_PER_MILE;
        String liveStats = "Time " + formatElapsedTime(elapsedMs)
                + " | Dist " + String.format(java.util.Locale.US, "%.2f mi", distanceMiles)
                + " | Avg " + formatPace(avgPaceMinPerMile);
        String splitText = mileSplitText == null || mileSplitText.trim().isEmpty()
                ? "Mile split: --"
                : mileSplitText;

        NotificationCompat.Builder builder = new NotificationCompat.Builder(this, CHANNEL_ID)
                .setContentTitle("LiveStride Run in Progress")
                .setContentText(liveStats)
                .setStyle(new NotificationCompat.BigTextStyle()
                        .bigText(liveStats + "\n" + splitText))
                .setSmallIcon(getApplicationInfo().icon)
                .setOngoing(true)
                .setOnlyAlertOnce(!alertThisUpdate)
                .setCategory(NotificationCompat.CATEGORY_SERVICE)
                .setVisibility(NotificationCompat.VISIBILITY_PUBLIC)
                .setPriority(alertThisUpdate
                        ? NotificationCompat.PRIORITY_MAX
                        : NotificationCompat.PRIORITY_HIGH)
                .setColor(0xFFD32F2F)
                .setColorized(!shouldRequestPromotedLiveUpdate())
                .setWhen(runStartTimeMs > 0L ? runStartTimeMs : System.currentTimeMillis())
                .setShowWhen(true)
                .setUsesChronometer(runStartTimeMs > 0L)
                .addAction(
                        android.R.drawable.ic_media_pause,
                        "STOP RUN",
                        stopPendingIntent
                );

        if (alertThisUpdate) {
            builder.setDefaults(NotificationCompat.DEFAULT_VIBRATE)
                    .setVibrate(new long[]{0L, 180L, 90L, 180L});
        }

        if (shouldRequestPromotedLiveUpdate()) {
            builder.getExtras().putBoolean(EXTRA_REQUEST_PROMOTED_ONGOING, true);
            builder.getExtras().putString(
                    EXTRA_SHORT_CRITICAL_TEXT,
                    formatLiveUpdateChipText(distanceMiles)
            );
        }

        return builder.build();
    }

    private static boolean shouldRequestPromotedLiveUpdate() {
        return Build.VERSION.SDK_INT >= ANDROID_16_API_LEVEL;
    }

    private static String formatLiveUpdateChipText(double distanceMiles) {
        return String.format(java.util.Locale.US, "%.2f mi", Math.max(0.0, distanceMiles));
    }

    private static String formatElapsedTime(long elapsedMs) {
        long totalSeconds = Math.max(0L, Math.round(elapsedMs / 1000.0));
        long hours = totalSeconds / 3600L;
        long minutes = (totalSeconds % 3600L) / 60L;
        long seconds = totalSeconds % 60L;

        if (hours > 0L) {
            return String.format(java.util.Locale.US, "%d:%02d:%02d", hours, minutes, seconds);
        }

        return String.format(java.util.Locale.US, "%d:%02d", minutes, seconds);
    }

    private static String formatPace(double paceMinPerMile) {
        if (!Double.isFinite(paceMinPerMile) || paceMinPerMile <= 0.0) {
            return "--/mi";
        }

        long totalSeconds = Math.round(paceMinPerMile * 60.0);
        long minutes = totalSeconds / 60L;
        long seconds = totalSeconds % 60L;
        return String.format(java.util.Locale.US, "%d:%02d/mi", minutes, seconds);
    }

    private void createNotificationChannel() {
        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
            NotificationChannel channel = new NotificationChannel(
                    CHANNEL_ID,
                    "LiveStride Run Controls",
                    NotificationManager.IMPORTANCE_HIGH
            );
            channel.setLockscreenVisibility(Notification.VISIBILITY_PUBLIC);
            channel.enableVibration(true);
            channel.setVibrationPattern(new long[]{0L, 180L, 90L, 180L});

            NotificationManager manager = getSystemService(NotificationManager.class);
            if (manager != null) {
                manager.createNotificationChannel(channel);
            }
        }
    }

    private static String isoNow() {
        java.text.SimpleDateFormat formatter = new java.text.SimpleDateFormat(
                "yyyy-MM-dd'T'HH:mm:ss.SSS'Z'",
                java.util.Locale.US
        );
        formatter.setTimeZone(java.util.TimeZone.getTimeZone("UTC"));
        return formatter.format(new java.util.Date());
    }

    private void saveNativeTrackerState(boolean running, long stoppedAtMs, long finalTimeMs) {
        SharedPreferences.Editor editor = getSharedPreferences(STATE_PREFS_NAME, MODE_PRIVATE)
                .edit()
                .putBoolean(STATE_RUNNING, running)
                .putLong(STATE_STOPPED_AT_MS, stoppedAtMs)
                .putLong(STATE_FINAL_TIME_MS, finalTimeMs)
                .putLong(STATE_RUN_START_TIME_MS, runStartTimeMs)
                .putBoolean(STATE_LOCAL_ONLY, localOnly);

        if (runId != null) {
            editor.putString(STATE_RUN_ID, runId);
        }

        putStringPreferenceIfPresent(editor, STATE_SUPABASE_URL, supabaseUrl);
        putStringPreferenceIfPresent(editor, STATE_SUPABASE_ANON_KEY, supabaseAnonKey);
        putStringPreferenceIfPresent(editor, STATE_RUNNER_NAME, runnerName);
        putStringPreferenceIfPresent(editor, STATE_NATIVE_SESSION_ID, nativeSessionId);
        putStringPreferenceIfPresent(editor, STATE_FIREBASE_DATABASE_URL, firebaseDatabaseUrl);
        putStringPreferenceIfPresent(editor, STATE_FIREBASE_RUNNER_PATH, firebaseRunnerPath);

        editor.apply();
    }

    private void saveNativeRunConfig() {
        SharedPreferences.Editor editor = getSharedPreferences(STATE_PREFS_NAME, MODE_PRIVATE).edit();
        putStringPreferenceIfPresent(editor, STATE_RUN_ID, runId);
        putStringPreferenceIfPresent(editor, STATE_SUPABASE_URL, supabaseUrl);
        putStringPreferenceIfPresent(editor, STATE_SUPABASE_ANON_KEY, supabaseAnonKey);
        putStringPreferenceIfPresent(editor, STATE_RUNNER_NAME, runnerName);
        putStringPreferenceIfPresent(editor, STATE_NATIVE_SESSION_ID, nativeSessionId);
        putStringPreferenceIfPresent(editor, STATE_FIREBASE_DATABASE_URL, firebaseDatabaseUrl);
        putStringPreferenceIfPresent(editor, STATE_FIREBASE_RUNNER_PATH, firebaseRunnerPath);
        if (runStartTimeMs > 0L) {
            editor.putLong(STATE_RUN_START_TIME_MS, runStartTimeMs);
        }
        editor.putBoolean(STATE_LOCAL_ONLY, localOnly);
        editor.apply();
    }

    private void restoreNativeRunConfig(Intent intent) {
        SharedPreferences prefs = getSharedPreferences(STATE_PREFS_NAME, MODE_PRIVATE);

        supabaseUrl = firstNonBlank(
                intent.getStringExtra("supabaseUrl"),
                firstNonBlank(supabaseUrl, prefs.getString(STATE_SUPABASE_URL, null))
        );
        supabaseAnonKey = firstNonBlank(
                intent.getStringExtra("supabaseAnonKey"),
                firstNonBlank(supabaseAnonKey, prefs.getString(STATE_SUPABASE_ANON_KEY, null))
        );
        runId = firstNonBlank(
                intent.getStringExtra("runId"),
                firstNonBlank(runId, prefs.getString(STATE_RUN_ID, null))
        );
        runnerName = firstNonBlank(
                intent.getStringExtra("runnerName"),
                firstNonBlank(runnerName, prefs.getString(STATE_RUNNER_NAME, null))
        );
        nativeSessionId = firstNonBlank(
                intent.getStringExtra("nativeSessionId"),
                firstNonBlank(nativeSessionId, prefs.getString(STATE_NATIVE_SESSION_ID, null))
        );
        firebaseDatabaseUrl = firstNonBlank(
                intent.getStringExtra("firebaseDatabaseUrl"),
                firstNonBlank(firebaseDatabaseUrl, prefs.getString(STATE_FIREBASE_DATABASE_URL, null))
        );
        firebaseRunnerPath = firstNonBlank(
                intent.getStringExtra("firebaseRunnerPath"),
                firstNonBlank(firebaseRunnerPath, prefs.getString(STATE_FIREBASE_RUNNER_PATH, null))
        );

        if (runStartTimeMs <= 0L) {
            runStartTimeMs = prefs.getLong(STATE_RUN_START_TIME_MS, 0L);
        }
        localOnly = intent.getBooleanExtra(
                "localOnly",
                localOnly || prefs.getBoolean(STATE_LOCAL_ONLY, false)
        );
    }

    private static String firstNonBlank(String preferred, String fallback) {
        if (preferred != null && !preferred.trim().isEmpty()) {
            return preferred;
        }
        return fallback;
    }

    private static void putStringPreferenceIfPresent(
            SharedPreferences.Editor editor,
            String key,
            String value
    ) {
        if (value != null && !value.trim().isEmpty()) {
            editor.putString(key, value);
        }
    }

    private static void putStringExtraIfPresent(Intent intent, String key, String value) {
        if (value != null && !value.trim().isEmpty()) {
            intent.putExtra(key, value);
        }
    }
