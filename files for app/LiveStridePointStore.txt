package com.kdidso.livestride;

import android.content.ContentValues;
import android.content.Context;
import android.database.Cursor;
import android.database.sqlite.SQLiteDatabase;
import android.database.sqlite.SQLiteOpenHelper;
import android.util.Log;

import org.json.JSONException;
import org.json.JSONArray;
import org.json.JSONObject;

import java.io.File;
import java.io.FileInputStream;
import java.io.FileOutputStream;
import java.io.IOException;
import java.util.ArrayList;
import java.util.List;

/**
 * Local persistent storage for native LiveStride GPS points.
 *
 * Purpose:
 * - Save every native GPS point immediately on the phone.
 * - Mark points uploaded only after Supabase confirms success.
 * - Leave failed/offline points pending for a later flush phase.
 *
 * No external storage permission is needed. Android stores this database in:
 * /data/user/0/com.kdidso.livestride/databases/livestride_native_points.db
 */
public class LiveStridePointStore extends SQLiteOpenHelper {

    private static final String TAG = "LiveStrideSQLite";
    private static final String DATABASE_NAME = "livestride_native_points.db";
    private static final int DATABASE_VERSION = 2;

    public static final String TABLE_NATIVE_POINTS = "native_run_points";

    private final Context appContext;

    public LiveStridePointStore(Context context) {
        super(context, DATABASE_NAME, null, DATABASE_VERSION);
        this.appContext = context.getApplicationContext();

        File dbFile = this.appContext.getDatabasePath(DATABASE_NAME);
        File debugDir = getDebugExportDir();

        Log.i(TAG, "LiveStride SQLite store initialized. Internal db="
                + dbFile.getAbsolutePath()
                + " debugExportDir=" + debugDir.getAbsolutePath());
    }

    @Override
    public void onCreate(SQLiteDatabase db) {
        db.execSQL(
                "CREATE TABLE IF NOT EXISTS " + TABLE_NATIVE_POINTS + " (" +
                        "local_id INTEGER PRIMARY KEY AUTOINCREMENT, " +
                        "run_id TEXT, " +
                        "runner_name TEXT, " +
                        "source TEXT, " +
                        "native_session_id TEXT, " +
                        "sequence_num INTEGER, " +
                        "lat REAL NOT NULL, " +
                        "lon REAL NOT NULL, " +
                        "accuracy REAL, " +
                        "recorded_at TEXT NOT NULL, " +
                        "distance_m REAL, " +
                        "inst_pace_min_per_mile REAL, " +
                        "avg_pace_min_per_mile REAL, " +
                        "elevation_m REAL, " +
                        "app_state TEXT, " +
                        "note TEXT, " +
                        "uploaded INTEGER NOT NULL DEFAULT 0, " +
                        "upload_status TEXT NOT NULL DEFAULT 'pending', " +
                        "upload_attempts INTEGER NOT NULL DEFAULT 0, " +
                        "last_upload_error TEXT, " +
                        "last_attempt_at TEXT, " +
                        "last_attempt_at_ms INTEGER, " +
                        "uploaded_at TEXT, " +
                        "created_at_ms INTEGER NOT NULL" +
                        ")"
        );

        db.execSQL(
                "CREATE INDEX IF NOT EXISTS idx_native_points_run_sequence " +
                        "ON " + TABLE_NATIVE_POINTS + " (run_id, sequence_num)"
        );

        db.execSQL(
                "CREATE INDEX IF NOT EXISTS idx_native_points_uploaded " +
                        "ON " + TABLE_NATIVE_POINTS + " (uploaded, created_at_ms)"
        );
    }

    @Override
    public void onUpgrade(SQLiteDatabase db, int oldVersion, int newVersion) {
        if (oldVersion < 2) {
            addColumnIfMissing(db, TABLE_NATIVE_POINTS, "upload_status", "TEXT NOT NULL DEFAULT 'pending'");
            addColumnIfMissing(db, TABLE_NATIVE_POINTS, "upload_attempts", "INTEGER NOT NULL DEFAULT 0");
            addColumnIfMissing(db, TABLE_NATIVE_POINTS, "last_upload_error", "TEXT");
            addColumnIfMissing(db, TABLE_NATIVE_POINTS, "last_attempt_at", "TEXT");
            addColumnIfMissing(db, TABLE_NATIVE_POINTS, "last_attempt_at_ms", "INTEGER");
            db.execSQL("UPDATE " + TABLE_NATIVE_POINTS
                    + " SET upload_status = CASE WHEN uploaded = 1 THEN 'uploaded' ELSE 'pending' END"
                    + " WHERE upload_status IS NULL OR upload_status = ''");
        }
    }

    public synchronized long insertNativePoint(JSONObject body) throws JSONException {
        ContentValues values = new ContentValues();

        putString(values, "run_id", body, "run_id");
        putString(values, "runner_name", body, "runner_name");
        putString(values, "source", body, "source");
        putString(values, "native_session_id", body, "native_session_id");
        putInteger(values, "sequence_num", body, "sequence_num");
        putDouble(values, "lat", body, "lat");
        putDouble(values, "lon", body, "lon");
        putDouble(values, "accuracy", body, "accuracy");
        putString(values, "recorded_at", body, "recorded_at");
        putDouble(values, "distance_m", body, "distance_m");
        putDouble(values, "inst_pace_min_per_mile", body, "inst_pace_min_per_mile");
        putDouble(values, "avg_pace_min_per_mile", body, "avg_pace_min_per_mile");
        putDouble(values, "elevation_m", body, "elevation_m");
        putString(values, "app_state", body, "app_state");
        putString(values, "note", body, "note");

        values.put("uploaded", 0);
        values.put("upload_status", "pending");
        values.put("created_at_ms", System.currentTimeMillis());

        SQLiteDatabase db = getWritableDatabase();
        long localId = db.insertOrThrow(TABLE_NATIVE_POINTS, null, values);

        Log.i(TAG, "Saved native point local_id=" + localId
                + " seq=" + body.optInt("sequence_num", -1)
                + " pendingCount=" + countPending());

        exportDebugCopy();

        return localId;
    }

    public synchronized void markUploaded(long localId) {
        ContentValues values = new ContentValues();
        values.put("uploaded", 1);
        values.put("upload_status", "uploaded");
        values.put("uploaded_at", isoNow());

        SQLiteDatabase db = getWritableDatabase();
        db.update(
                TABLE_NATIVE_POINTS,
                values,
                "local_id = ?",
                new String[]{String.valueOf(localId)}
        );

        Log.i(TAG, "Marked native point uploaded local_id=" + localId
                + " pendingCount=" + countPending());

        exportDebugCopy();
    }

    public synchronized void markUploadInProgress(long localId) {
        ContentValues values = new ContentValues();
        values.put("upload_status", "uploading");
        values.put("last_attempt_at", isoNow());
        values.put("last_attempt_at_ms", System.currentTimeMillis());

        SQLiteDatabase db = getWritableDatabase();
        db.update(
                TABLE_NATIVE_POINTS,
                values,
                "local_id = ? AND uploaded = 0",
                new String[]{String.valueOf(localId)}
        );
    }

    public synchronized void updateElevation(long localId, double elevationMeters) {
        ContentValues values = new ContentValues();
        values.put("elevation_m", elevationMeters);

        SQLiteDatabase db = getWritableDatabase();
        db.update(
                TABLE_NATIVE_POINTS,
                values,
                "local_id = ?",
                new String[]{String.valueOf(localId)}
        );

        Log.i(TAG, "Updated native point elevation local_id=" + localId
                + " elevation_m=" + elevationMeters);

        exportDebugCopy();
    }

    public synchronized void markUploadFailed(long localId, String error) {
        ContentValues values = new ContentValues();
        values.put("uploaded", 0);
        values.put("upload_status", "pending");
        values.put("last_upload_error", error == null ? "Upload failed" : error);
        values.put("last_attempt_at", isoNow());
        values.put("last_attempt_at_ms", System.currentTimeMillis());

        SQLiteDatabase db = getWritableDatabase();
        db.update(
                TABLE_NATIVE_POINTS,
                values,
                "local_id = ?",
                new String[]{String.valueOf(localId)}
        );

        db.execSQL(
                "UPDATE " + TABLE_NATIVE_POINTS
                        + " SET upload_attempts = upload_attempts + 1 WHERE local_id = ?",
                new Object[]{localId}
        );

        Log.w(TAG, "Marked native point pending local_id=" + localId
                + " error=" + values.getAsString("last_upload_error")
                + " pendingCount=" + countPending());

        exportDebugCopy();
    }

    public synchronized int resetStaleUploadAttempts(long staleBeforeMs) {
        ContentValues values = new ContentValues();
        values.put("upload_status", "pending");

        SQLiteDatabase db = getWritableDatabase();
        int rows = db.update(
                TABLE_NATIVE_POINTS,
                values,
                "uploaded = 0 AND upload_status = ? AND last_attempt_at_ms < ?",
                new String[]{"uploading", String.valueOf(staleBeforeMs)}
        );

        if (rows > 0) {
            Log.w(TAG, "Reset stale native point uploads rows=" + rows);
            exportDebugCopy();
        }

        return rows;
    }

    public synchronized int countPending() {
        SQLiteDatabase db = getReadableDatabase();
        Cursor cursor = db.rawQuery(
                "SELECT COUNT(*) FROM " + TABLE_NATIVE_POINTS + " WHERE uploaded = 0",
                null
        );

        try {
            if (cursor.moveToFirst()) {
                return cursor.getInt(0);
            }
            return 0;
        } finally {
            cursor.close();
        }
    }

    public synchronized int countPendingForRun(String runId) {
        if (runId == null) return 0;

        SQLiteDatabase db = getReadableDatabase();
        Cursor cursor = db.rawQuery(
                "SELECT COUNT(*) FROM " + TABLE_NATIVE_POINTS + " WHERE uploaded = 0 AND run_id = ?",
                new String[]{runId}
        );

        try {
            if (cursor.moveToFirst()) {
                return cursor.getInt(0);
            }
            return 0;
        } finally {
            cursor.close();
        }
    }

    public synchronized JSONObject getRunSummaryFromNativePoints(String runId) {
        if (runId == null) return null;

        SQLiteDatabase db = getReadableDatabase();
        Cursor cursor = db.rawQuery(
                "SELECT lat, lon, recorded_at, distance_m FROM " + TABLE_NATIVE_POINTS
                        + " WHERE run_id = ?"
                        + " ORDER BY sequence_num ASC, recorded_at ASC, local_id ASC",
                new String[]{runId}
        );

        try {
            if (!cursor.moveToFirst()) {
                return null;
            }

            String firstRecordedAt = cursor.getString(cursor.getColumnIndexOrThrow("recorded_at"));
            String lastRecordedAt = firstRecordedAt;
            double latestLat = cursor.getDouble(cursor.getColumnIndexOrThrow("lat"));
            double latestLon = cursor.getDouble(cursor.getColumnIndexOrThrow("lon"));
            double totalDistanceM = getNullableDouble(cursor, "distance_m", 0.0);

            while (cursor.moveToNext()) {
                latestLat = cursor.getDouble(cursor.getColumnIndexOrThrow("lat"));
                latestLon = cursor.getDouble(cursor.getColumnIndexOrThrow("lon"));
                lastRecordedAt = cursor.getString(cursor.getColumnIndexOrThrow("recorded_at"));
                totalDistanceM = getNullableDouble(cursor, "distance_m", totalDistanceM);
            }

            JSONObject summary = new JSONObject();
            summary.put("latest_lat", latestLat);
            summary.put("latest_lon", latestLon);
            summary.put("latest_timestamp", lastRecordedAt);
            summary.put("total_distance_m", totalDistanceM);

            Long firstMs = parseIsoMillis(firstRecordedAt);
            Long lastMs = parseIsoMillis(lastRecordedAt);
            if (firstMs != null && lastMs != null && lastMs >= firstMs) {
                summary.put("final_time_ms", lastMs - firstMs);
            }

            return summary;
        } catch (JSONException e) {
            Log.w(TAG, "Could not build native run summary: " + e.getMessage());
            return null;
        } finally {
            cursor.close();
        }
    }

    public synchronized JSONObject getRunDisplaySnapshot(
            String runId,
            int afterSequenceNum,
            int beforeSequenceNum,
            boolean includeFullPath
    ) {
        if (runId == null) return null;

        SQLiteDatabase db = getReadableDatabase();
        JSONArray points = new JSONArray();
        JSONObject latestPoint = null;
        String firstRecordedAt = null;
        int latestSequenceNum = -1;

        String pointWhere;
        String[] pointArgs;

        if (includeFullPath) {
            pointWhere = "run_id = ?";
            pointArgs = new String[]{runId};
        } else if (beforeSequenceNum >= 0) {
            pointWhere = "run_id = ? AND sequence_num > ? AND sequence_num < ?";
            pointArgs = new String[]{
                    runId,
                    String.valueOf(afterSequenceNum),
                    String.valueOf(beforeSequenceNum)
            };
        } else {
            pointWhere = "run_id = ? AND sequence_num > ?";
            pointArgs = new String[]{runId, String.valueOf(afterSequenceNum)};
        }

        Cursor pointCursor = db.query(
                TABLE_NATIVE_POINTS,
                null,
                pointWhere,
                pointArgs,
                null,
                null,
                "sequence_num ASC, recorded_at ASC, local_id ASC"
        );

        try {
            while (pointCursor.moveToNext()) {
                points.put(cursorToJson(pointCursor));
            }
        } finally {
            pointCursor.close();
        }

        Cursor latestCursor = db.query(
                TABLE_NATIVE_POINTS,
                null,
                "run_id = ?",
                new String[]{runId},
                null,
                null,
                "sequence_num DESC, recorded_at DESC, local_id DESC",
                "1"
        );

        try {
            if (latestCursor.moveToFirst()) {
                latestPoint = cursorToJson(latestCursor);
                int sequenceIdx = latestCursor.getColumnIndex("sequence_num");
                if (sequenceIdx >= 0 && !latestCursor.isNull(sequenceIdx)) {
                    latestSequenceNum = latestCursor.getInt(sequenceIdx);
                }
            }
        } finally {
            latestCursor.close();
        }

        Cursor firstCursor = db.query(
                TABLE_NATIVE_POINTS,
                new String[]{"recorded_at"},
                "run_id = ?",
                new String[]{runId},
                null,
                null,
                "sequence_num ASC, recorded_at ASC, local_id ASC",
                "1"
        );

        try {
            if (firstCursor.moveToFirst()) {
                firstRecordedAt = firstCursor.getString(0);
            }
        } finally {
            firstCursor.close();
        }

        try {
            JSONObject snapshot = new JSONObject();
            snapshot.put("run_id", runId);
            snapshot.put("points", points);
            snapshot.put("latestPoint", latestPoint == null ? JSONObject.NULL : latestPoint);
            snapshot.put("latestSequenceNum", latestSequenceNum);
            snapshot.put("firstRecordedAt", firstRecordedAt == null ? JSONObject.NULL : firstRecordedAt);
            snapshot.put("pendingCount", countPendingForRun(runId));
            snapshot.put("includeFullPath", includeFullPath);
            return snapshot;
        } catch (JSONException e) {
            Log.w(TAG, "Could not build native run display snapshot: " + e.getMessage());
            return null;
        }
    }

    public synchronized List<JSONObject> getPendingUploadBodies(int limit) {
        List<JSONObject> rows = new ArrayList<>();

        SQLiteDatabase db = getReadableDatabase();
        Cursor cursor = db.query(
                TABLE_NATIVE_POINTS,
                null,
                "uploaded = 0 AND upload_status = ?",
                new String[]{"pending"},
                null,
                null,
                "created_at_ms ASC, local_id ASC",
                String.valueOf(limit)
        );

        try {
            while (cursor.moveToNext()) {
                rows.add(cursorToJson(cursor));
            }
        } finally {
            cursor.close();
        }

        return rows;
    }

    private JSONObject cursorToJson(Cursor cursor) {
        JSONObject json = new JSONObject();

        try {
            putJsonLong(json, "local_id", cursor);
            putJsonString(json, "run_id", cursor);
            putJsonString(json, "runner_name", cursor);
            putJsonString(json, "source", cursor);
            putJsonString(json, "native_session_id", cursor);
            putJsonInt(json, "sequence_num", cursor);
            putJsonDouble(json, "lat", cursor);
            putJsonDouble(json, "lon", cursor);
            putJsonDouble(json, "accuracy", cursor);
            putJsonString(json, "recorded_at", cursor);
            putJsonDouble(json, "distance_m", cursor);
            putJsonDouble(json, "inst_pace_min_per_mile", cursor);
            putJsonDouble(json, "avg_pace_min_per_mile", cursor);
            putJsonDouble(json, "elevation_m", cursor);
            putJsonString(json, "app_state", cursor);
            putJsonString(json, "note", cursor);
        } catch (JSONException e) {
            Log.w(TAG, "Could not build pending point JSON: " + e.getMessage());
        }

        return json;
    }

    private static void putString(ContentValues values, String column, JSONObject body, String key) {
        if (!body.has(key) || body.isNull(key)) {
            values.putNull(column);
        } else {
            values.put(column, body.optString(key, null));
        }
    }

    private static void putInteger(ContentValues values, String column, JSONObject body, String key) {
        if (!body.has(key) || body.isNull(key)) {
            values.putNull(column);
        } else {
            values.put(column, body.optInt(key));
        }
    }

    private static void putDouble(ContentValues values, String column, JSONObject body, String key) {
        if (!body.has(key) || body.isNull(key)) {
            values.putNull(column);
        } else {
            values.put(column, body.optDouble(key));
        }
    }

    private static void putJsonString(JSONObject json, String key, Cursor cursor) throws JSONException {
        int idx = cursor.getColumnIndex(key);
        if (idx < 0 || cursor.isNull(idx)) {
            json.put(key, JSONObject.NULL);
        } else {
            json.put(key, cursor.getString(idx));
        }
    }

    private static void putJsonInt(JSONObject json, String key, Cursor cursor) throws JSONException {
        int idx = cursor.getColumnIndex(key);
        if (idx < 0 || cursor.isNull(idx)) {
            json.put(key, JSONObject.NULL);
        } else {
            json.put(key, cursor.getInt(idx));
        }
    }

    private static void putJsonLong(JSONObject json, String key, Cursor cursor) throws JSONException {
        int idx = cursor.getColumnIndex(key);
        if (idx < 0 || cursor.isNull(idx)) {
            json.put(key, JSONObject.NULL);
        } else {
            json.put(key, cursor.getLong(idx));
        }
    }

    private static void putJsonDouble(JSONObject json, String key, Cursor cursor) throws JSONException {
        int idx = cursor.getColumnIndex(key);
        if (idx < 0 || cursor.isNull(idx)) {
            json.put(key, JSONObject.NULL);
        } else {
            json.put(key, cursor.getDouble(idx));
        }
    }

    private static double getNullableDouble(Cursor cursor, String column, double fallback) {
        int idx = cursor.getColumnIndex(column);
        if (idx < 0 || cursor.isNull(idx)) {
            return fallback;
        }

        return cursor.getDouble(idx);
    }

    private static Long parseIsoMillis(String iso) {
        if (iso == null || iso.trim().isEmpty()) {
            return null;
        }

        try {
            java.text.SimpleDateFormat formatter =
                    new java.text.SimpleDateFormat(
                            "yyyy-MM-dd'T'HH:mm:ss.SSS'Z'",
                            java.util.Locale.US
                    );
            formatter.setTimeZone(java.util.TimeZone.getTimeZone("UTC"));
            java.util.Date parsed = formatter.parse(iso);
            return parsed == null ? null : parsed.getTime();
        } catch (Exception e) {
            Log.w(TAG, "Could not parse native point timestamp: " + iso);
            return null;
        }
    }

    private static void addColumnIfMissing(
            SQLiteDatabase db,
            String table,
            String column,
            String definition
    ) {
        Cursor cursor = db.rawQuery("PRAGMA table_info(" + table + ")", null);

        try {
            while (cursor.moveToNext()) {
                int nameIndex = cursor.getColumnIndex("name");
                if (nameIndex >= 0 && column.equals(cursor.getString(nameIndex))) {
                    return;
                }
            }
        } finally {
            cursor.close();
        }

        db.execSQL("ALTER TABLE " + table + " ADD COLUMN " + column + " " + definition);
    }

    public synchronized String exportDebugCopy() {
        try {
            // Make sure the database file has been created/opened.
            getReadableDatabase();

            File dbFile = appContext.getDatabasePath(DATABASE_NAME);
            File debugDir = getDebugExportDir();

            if (!debugDir.exists() && !debugDir.mkdirs()) {
                Log.w(TAG, "Could not create SQLite debug export directory: "
                        + debugDir.getAbsolutePath());
                return null;
            }

            File copiedDb = new File(debugDir, DATABASE_NAME);
            copyFileIfExists(dbFile, copiedDb);

            // Copy common companion files if Android creates them.
            copyFileIfExists(
                    new File(dbFile.getAbsolutePath() + "-journal"),
                    new File(debugDir, DATABASE_NAME + "-journal")
            );
            copyFileIfExists(
                    new File(dbFile.getAbsolutePath() + "-wal"),
                    new File(debugDir, DATABASE_NAME + "-wal")
            );
            copyFileIfExists(
                    new File(dbFile.getAbsolutePath() + "-shm"),
                    new File(debugDir, DATABASE_NAME + "-shm")
            );

            Log.i(TAG, "SQLite debug copy exported to "
                    + copiedDb.getAbsolutePath()
                    + " pendingCount=" + countPending());

            return copiedDb.getAbsolutePath();

        } catch (Exception e) {
            Log.w(TAG, "SQLite debug export failed: " + e.getMessage());
            return null;
        }
    }

    private File getDebugExportDir() {
        File externalFiles = appContext.getExternalFilesDir(null);

        if (externalFiles != null) {
            return new File(externalFiles, "livestride_sqlite_debug");
        }

        return new File(appContext.getFilesDir(), "livestride_sqlite_debug");
    }

    private static void copyFileIfExists(File source, File destination) throws IOException {
        if (source == null || !source.exists()) {
            return;
        }

        try (FileInputStream in = new FileInputStream(source);
             FileOutputStream out = new FileOutputStream(destination)) {

            byte[] buffer = new byte[8192];
            int read;

            while ((read = in.read(buffer)) != -1) {
                out.write(buffer, 0, read);
            }

            out.flush();
        }
    }

    private static String isoNow() {
        java.text.SimpleDateFormat formatter = new java.text.SimpleDateFormat(
                "yyyy-MM-dd'T'HH:mm:ss.SSS'Z'",
                java.util.Locale.US
        );
        formatter.setTimeZone(java.util.TimeZone.getTimeZone("UTC"));
        return formatter.format(new java.util.Date());
    }
}
