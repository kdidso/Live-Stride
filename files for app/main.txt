function formatProfileActivity(run) {
  const started = new Date(run.started_at);
  const miles = Number(run.distance_m || 0) / 1609.344;
  const elapsedMs = Number(run.elapsed_time_ms || 0);
  const totalSeconds = Math.max(0, Math.round(elapsedMs / 1000));
  const hours = Math.floor(totalSeconds / 3600);
  const minutes = Math.floor((totalSeconds % 3600) / 60);
  const seconds = totalSeconds % 60;
  const duration = hours > 0
    ? `${hours}:${String(minutes).padStart(2, "0")}:${String(seconds).padStart(2, "0")}`
    : `${minutes}:${String(seconds).padStart(2, "0")}`;

  const item = document.createElement("article");
  item.className = "profile-activity";
  const title = document.createElement("strong");
  title.textContent = run.title || started.toLocaleString();
  const details = document.createElement("span");
  details.textContent = `${miles.toFixed(2)} mi | ${duration} | ${run.source}`;
  item.append(title, details);
  return item;
}

async function loadProfileActivities() {
  if (!profileActivities || !profilesSession?.user) return;
  profileActivities.innerHTML = '<div class="profile-empty">Loading activities...</div>';

  const { data, error } = await profilesSupabaseClient
    .from("saved_runs")
    .select("id, title, source, started_at, elapsed_time_ms, distance_m, archive_status")
    .order("started_at", { ascending: false })
    .limit(50);

  profileActivities.innerHTML = "";
  if (error) {
    const message = document.createElement("div");
    message.className = "profile-empty";
    message.textContent = "Activities could not be loaded right now.";
    profileActivities.appendChild(message);
    console.warn("Profile activity load failed:", error);
    return;
  }

  if (!data?.length) {
    const empty = document.createElement("div");
    empty.className = "profile-empty";
    empty.textContent = "No saved runs yet.";
    profileActivities.appendChild(empty);
    return;
  }

  data.forEach(run => profileActivities.appendChild(formatProfileActivity(run)));
}

async function openProfileView() {

  profileView.style.display = "block";
  profileView.setAttribute("aria-hidden", "false");

}

function applyProfilesSession(session) {
  profilesSession = session || null;
  profileBtn?.classList.toggle("signed-in", Boolean(profilesSession?.user));
}

async function submitProfileSignIn() {
  const email = profileAuthEmail?.value.trim() || "";
  const password = profileAuthPassword?.value || "";
  if (!email || !password) {
    setProfileAuthMessage("Enter both your email and password.");
    return;
  }

  setProfileAuthBusy(true);
  setProfileAuthMessage("Signing in...");
  const { data, error } = await profilesSupabaseClient.auth.signInWithPassword({ email, password });
  setProfileAuthBusy(false);

  if (error) {
    setProfileAuthMessage(error.message);
    return;
  }

  applyProfilesSession(data.session);
  profileAuthPassword.value = "";
  await openProfileView();
}

async function submitProfileCreateAccount() {
  const email = profileAuthEmail?.value.trim() || "";
  const password = profileAuthPassword?.value || "";
  if (!email || !password) {
    setProfileAuthMessage("Enter an email and a password of at least 8 characters.");
    return;
  }
  if (password.length < 8) {
    setProfileAuthMessage("Use a password of at least 8 characters.");
    return;
  }
  setProfileAuthBusy(true);
  setProfileAuthMessage("Creating account...");
  const { data, error } = await profilesSupabaseClient.auth.signUp({
    email,
    password,
    options: { emailRedirectTo: "https://app.livestride.run" }
  });
  setProfileAuthBusy(false);

  if (error) {
    setProfileAuthMessage(error.message);
    return;
  }

  profileAuthPassword.value = "";
  if (data.session) {
    applyProfilesSession(data.session);
    await openProfileView();
  } else {
    setProfileAuthMessage("Check your email to confirm your account, then return here to sign in.", true);
  }
}

async function submitProfilePasswordResetEmail() {
  const email = profileAuthEmail?.value.trim() || "";
  if (!email || !profileAuthEmail?.checkValidity()) {
    setProfileAuthMessage("Enter a valid email address.");
    profileAuthEmail?.focus();
    return;
  }

  setProfileAuthBusy(true);
  setProfileAuthMessage("Requesting a password reset...");
  const { error } = await profilesSupabaseClient.auth.resetPasswordForEmail(email, {
    redirectTo: "https://app.livestride.run"
  });
  setProfileAuthBusy(false);

  if (error) console.warn("Profile password reset request failed:", error);
  setProfileAuthMessage(
    "If that email belongs to a LiveStride account, a password reset link has been sent.",
    true
  );
}

async function submitProfileNewPassword() {
  const password = profileAuthPassword?.value || "";
  if (password.length < 8) {
    setProfileAuthMessage("Use a new password of at least 8 characters.");
    return;
  }

  setProfileAuthBusy(true);
  setProfileAuthMessage("Saving your new password...");
  const { error } = await profilesSupabaseClient.auth.updateUser({ password });
  setProfileAuthBusy(false);

  if (error) {
    setProfileAuthMessage("The reset link may have expired. Request a new password reset email and try again.");
    return;
  }

  setProfilePasswordRecoveryMode(false);
  setProfileAuthMessage("Your password has been updated. You are now signed in.", true);
}

async function initializeProfilesAuth() {
  profilesSupabaseClient.auth.onAuthStateChange((event, session) => {
    applyProfilesSession(session);
    if (event === "PASSWORD_RECOVERY") {
      setProfilePasswordRecoveryMode(true);
      openProfileAuth();
    }
  });

async function handleExportPageMessage(event) {
  if (event.origin !== "https://kdidso.github.io") return;

  const data = event.data || {};
  if (data.source !== "livestride-export") return;

  const replyToExportPage = (payload) => {
    if (!exportPageFrame?.contentWindow) return;
    exportPageFrame.contentWindow.postMessage(
      {
        source: "livestride-app",
        requestId: data.requestId || null,
        ...payload
      },
      "https://kdidso.github.io"
    );
  };

  if (data.type === "saveGpx") {
    const filename = String(data.filename || "LiveStride.gpx");
    const content = String(data.content || "");

    if (!content) {
      replyToExportPage({
        type: "saveGpxResult",
        ok: false,
        message: "No GPX content was provided."
      });
      return;
    }

    try {
      if (LiveStrideExport?.saveGpxToDownloads) {
        const result = await LiveStrideExport.saveGpxToDownloads({ filename, content });
        replyToExportPage({
          type: "saveGpxResult",
          ok: true,
          filename: result?.filename || filename
        });
      } else {
        downloadGpxInBrowser(filename, content);
        replyToExportPage({
          type: "saveGpxResult",
          ok: true,
          filename
        });
      }
    } catch (err) {
      replyToExportPage({
        type: "saveGpxResult",
        ok: false,
        message: err?.message || String(err)
      });
    }
  } else if (data.type === "openExternal" || data.type === "openInPanel") {
    const url = String(data.url || "");
    if (url.startsWith("https://www.strava.com/") || url.startsWith("https://strava.com/")) {
      try {
        await openExternalUrl(url);
      } catch (err) {
        console.warn("Unable to open external URL:", err);
        window.open(url, "_blank");
      }
    }
  }
}

function queueFirebaseFinish(finishPayload = null) {
  const fallbackPayload = {
    finished: true,
    finish_requested_at: new Date().toISOString(),
    finished_at: new Date().toISOString(),
    timestamp: Date.now()
  };

  localStorage.setItem(
    FIREBASE_FINISH_QUEUE_KEY,
    JSON.stringify(finishPayload || fallbackPayload)
  );
}

function clearFirebaseFinishQueue() {
  localStorage.removeItem(FIREBASE_FINISH_QUEUE_KEY);
}

function getQueuedFirebaseFinish() {
  const raw = localStorage.getItem(FIREBASE_FINISH_QUEUE_KEY);
  if (!raw) return null;

  try {
    return JSON.parse(raw);
  } catch (err) {
    return {
      finished: true,
      finish_requested_at: new Date().toISOString(),
      finished_at: new Date().toISOString(),
      timestamp: Date.now()
    };
  }
}

function hasQueuedFirebaseFinish() {
  return getQueuedFirebaseFinish() != null;
}

async function flushFirebaseFinish() {
  if (!broadcastRef) return true;
  const finishPayload = getQueuedFirebaseFinish();
  if (!finishPayload) return true;
  if (!navigator.onLine) return false;

  try {
    await broadcastRef.update(finishPayload);

    clearFirebaseFinishQueue();
    return true;
  } catch (err) {
    console.warn("Firebase finish retry failed:", err);
    return false;
  }
}

function loadSyncQueue() {
  try {
    return JSON.parse(localStorage.getItem(SYNC_QUEUE_KEY)) || {
      points: [],
      runUpdates: {},
      finishEvents: {}
    };
  } catch {
    return { points: [], runUpdates: {}, finishEvents: {} };
  }
}

function saveSyncQueue(queue) {
  localStorage.setItem(SYNC_QUEUE_KEY, JSON.stringify(queue));
}

function queueRunPoint(pointPayload) {
  const queue = loadSyncQueue();

  const exists = queue.points.some(p =>
    p.run_id === pointPayload.run_id &&
    p.sequence_num === pointPayload.sequence_num
  );

  if (!exists) {
    queue.points.push(pointPayload);
    saveSyncQueue(queue);
  }
}

function queueRunUpdate(runId, updatePayload) {
  const queue = loadSyncQueue();
  queue.runUpdates[runId] = {
    ...(queue.runUpdates[runId] || {}),
    ...updatePayload
  };
  saveSyncQueue(queue);
}

function queueFinishEvent(runId, finishPayload) {
  const queue = loadSyncQueue();
  queue.finishEvents[runId] = finishPayload;
  saveSyncQueue(queue);
}

function updateQueuedPointElevation(runId, sequenceNum, elevationM) {
  const queue = loadSyncQueue();
  let changed = false;

  queue.points.forEach(point => {
    if (point.run_id === runId && point.sequence_num === sequenceNum) {
      point.elevation_m = elevationM;
      changed = true;
    }
  });

  if (changed) saveSyncQueue(queue);
}

// -> Utility functions
function metersToMiles(meters) {
  return meters / 1609.34;
}

function haversineDistance(a, b) {
  const toRad = value => value * Math.PI / 180;
  const R = 6371000;
  const lat1 = toRad(a[1]);
  const lat2 = toRad(b[1]);
  const dLat = toRad(b[1] - a[1]);
  const dLng = toRad(b[0] - a[0]);
  const h = Math.sin(dLat / 2) ** 2 +
    Math.cos(lat1) * Math.cos(lat2) * Math.sin(dLng / 2) ** 2;
  return R * 2 * Math.atan2(Math.sqrt(h), Math.sqrt(1 - h));
}

function formatTime(ms) {
  const totalSec = Math.floor(ms / 1000);
  const h = String(Math.floor(totalSec / 3600)).padStart(2, "0");
  const m = String(Math.floor((totalSec % 3600) / 60)).padStart(2, "0");
  const s = String(totalSec % 60).padStart(2, "0");
  return `${h}:${m}:${s}`;
}

function formatPace(minsPerMile) {
  if (!isFinite(minsPerMile) || minsPerMile <= 0) return "-";
  let minutes = Math.floor(minsPerMile);
  let seconds = Math.round((minsPerMile - minutes) * 60);
  if (seconds === 60) {
    minutes += 1;
    seconds = 0;
  }
  return `${minutes}:${seconds.toString().padStart(2, "0")} /mi`;
}

function calculateMileSplits(points, options = {}) {
  const sorted = (Array.isArray(points) ? points : [])
    .filter(point =>
      Number.isFinite(point?.timestamp) &&
      Number.isFinite(point?.lat) &&
      Number.isFinite(point?.lon)
    )
    .slice()
    .sort((a, b) => {
      if (a.sequenceNum != null && b.sequenceNum != null) {
        return a.sequenceNum - b.sequenceNum;
      }

      return a.timestamp - b.timestamp;
    });

  if (sorted.length < 2) return [];

  const pointsWithDistance = [];

  sorted.forEach((point, index) => {
    if (Number.isFinite(point.distanceM)) {
      pointsWithDistance.push(point);
      return;
    }

    const previous = index > 0 ? sorted[index - 1] : null;
    const previousDistance = index > 0
      ? pointsWithDistance[index - 1].distanceM
      : 0;
    const segmentMeters = previous
      ? haversineDistance([previous.lon, previous.lat], [point.lon, point.lat])
      : 0;

    pointsWithDistance.push({
      ...point,
      distanceM: previousDistance + segmentMeters
    });
  });

  const interpolateTimeAtDistance = (targetDistanceM) => {
    if (targetDistanceM < 0) return null;

    for (let i = 1; i < pointsWithDistance.length; i++) {
      const previous = pointsWithDistance[i - 1];
      const current = pointsWithDistance[i];

      if (
        !Number.isFinite(previous.distanceM) ||
        !Number.isFinite(current.distanceM) ||
        !Number.isFinite(previous.timestamp) ||
        !Number.isFinite(current.timestamp) ||
        current.distanceM <= previous.distanceM ||
        current.timestamp <= previous.timestamp
      ) {
        continue;
      }

      if (targetDistanceM <= current.distanceM) {
        if (targetDistanceM <= previous.distanceM) {
          return previous.timestamp;
        }

        const ratio = (targetDistanceM - previous.distanceM) /
          (current.distanceM - previous.distanceM);
        return previous.timestamp + ratio * (current.timestamp - previous.timestamp);
      }
    }

    return null;
  };

  const splits = [];
  const mileMeters = 1609.34;
  const startPoint = pointsWithDistance[0];
  let previousBoundaryTime = startPoint.timestamp;
  let nextMile = 1;

  for (let i = 1; i < pointsWithDistance.length; i++) {
    const previous = pointsWithDistance[i - 1];
    const current = pointsWithDistance[i];

    if (
      !Number.isFinite(previous.distanceM) ||
      !Number.isFinite(current.distanceM) ||
      !Number.isFinite(previous.timestamp) ||
      !Number.isFinite(current.timestamp) ||
      current.distanceM <= previous.distanceM ||
      current.timestamp <= previous.timestamp
    ) {
      continue;
    }

    while (current.distanceM >= nextMile * mileMeters) {
      const boundaryDistance = nextMile * mileMeters;
      const ratio = (boundaryDistance - previous.distanceM) /
        (current.distanceM - previous.distanceM);

      if (ratio < 0 || ratio > 1) break;

      const boundaryTime = previous.timestamp +
        ratio * (current.timestamp - previous.timestamp);
      const splitDurationMs = boundaryTime - previousBoundaryTime;

      if (splitDurationMs > 0) {
        splits.push({
          key: `mile-${nextMile}`,
          distanceM: boundaryDistance,
          label: `Mile ${nextMile}`,
          value: formatPace(splitDurationMs / 60000),
          mile: nextMile,
          durationMs: splitDurationMs,
          pace: splitDurationMs / 60000
        });
      }

      previousBoundaryTime = boundaryTime;
      nextMile += 1;
    }
  }

  const finalPoint = pointsWithDistance[pointsWithDistance.length - 1];
  const totalDistanceM = finalPoint.distanceM;
  const totalElapsedMs = finalPoint.timestamp - startPoint.timestamp;

  [
    { label: "Half 13.1", distanceM: 13.1 * mileMeters, key: "half-marathon" },
    { label: "Marathon 26.2", distanceM: 26.2 * mileMeters, key: "marathon" }
  ].forEach(milestone => {
    if (totalDistanceM < milestone.distanceM) return;

    const milestoneTime = interpolateTimeAtDistance(milestone.distanceM);
    if (!Number.isFinite(milestoneTime)) return;

    splits.push({
      ...milestone,
      value: formatTime(Math.max(0, milestoneTime - startPoint.timestamp))
    });
  });

  if (options.includeFinalPartial && totalDistanceM > 0 && totalElapsedMs > 0) {
    const totalMiles = metersToMiles(totalDistanceM);
    const finalPartialMiles = totalMiles - Math.floor(totalMiles);

    if (finalPartialMiles >= 0.01) {
      const partialStartDistanceM = Math.floor(totalMiles) * mileMeters;
      const partialStartTime = interpolateTimeAtDistance(partialStartDistanceM);

      if (Number.isFinite(partialStartTime) && finalPoint.timestamp > partialStartTime) {
        const partialDurationMs = finalPoint.timestamp - partialStartTime;
        const partialPace = (partialDurationMs / 60000) / finalPartialMiles;

        splits.push({
          key: "finish-partial",
          distanceM: totalDistanceM + 0.001,
          label: `Finish +${finalPartialMiles.toFixed(2)} mi`,
          value: formatPace(partialPace)
        });
      }
    }
  }

  return splits.sort((a, b) => {
    if (a.distanceM !== b.distanceM) return a.distanceM - b.distanceM;
    return String(a.key).localeCompare(String(b.key));
  });
}

function updateSplitsPanel(points = null, options = {}) {
  const sourcePoints = points || (
    document.body.classList.contains("viewer-mode") && nativePathCoords.length > 0
      ? nativePathCoords
      : pathCoords
  );
  const includeFinalPartial = options.includeFinalPartial ?? (
    document.body.classList.contains("viewer-mode")
      ? currentViewerRunIsLive === false
      : stopBtn?.disabled === true
  );
  const splits = calculateMileSplits(sourcePoints, { includeFinalPartial });

  if (splits.length === 0) {
    hideSplitsPanel();
    return;
  }

  ensureSplitsPanel();
  positionSplitsPanel();

  const signature = splits
    .map(split => `${split.key}:${split.value}`)
    .join("|");

  if (signature === lastSplitsSignature) {
    splitsPanelEl.style.display = "block";
    return;
  }

  const shouldScrollToLatest = splits.length > (lastSplitsSignature ? lastSplitsSignature.split("|").length : 0);
  splitsListEl.innerHTML = "";

  splits.forEach(split => {
    const row = document.createElement("div");
    row.style.display = "flex";
    row.style.justifyContent = "space-between";
    row.style.gap = "10px";
    row.style.padding = "6px 10px";
    row.style.fontSize = "13px";
    row.style.borderBottom = "1px solid rgba(0,0,0,0.08)";
    row.innerHTML = `<span>${split.label}</span><strong>${split.value}</strong>`;
    splitsListEl.appendChild(row);
  });

  splitsPanelEl.style.display = "block";
  lastSplitsSignature = signature;

  if (shouldScrollToLatest) {
    splitsListEl.scrollTop = splitsListEl.scrollHeight;
  }
}

function pointInPolygon(lon, lat, polygon) {
  let inside = false;
  for (let i = 0, j = polygon.length - 1; i < polygon.length; j = i++) {
    const xi = polygon[i][0];
    const yi = polygon[i][1];
    const xj = polygon[j][0];
    const yj = polygon[j][1];
    const intersects = ((yi > lat) !== (yj > lat)) &&
      lon < ((xj - xi) * (lat - yi)) / (yj - yi) + xi;
    if (intersects) inside = !inside;
  }
  return inside;
}

function isWithin50StateElevationCoverage(lat, lon) {
  if (!Number.isFinite(lat) || !Number.isFinite(lon)) return true;
  return FIFTY_STATE_ELEVATION_GUARD_AREAS.some(area =>
    pointInPolygon(lon, lat, area)
  );
}

function shouldUseArcGISElevationFallback(lat, lon) {
  return Number.isFinite(lat) &&
    Number.isFinite(lon) &&
    lat >= -85.05 &&
    lat <= 85.05 &&
    lon >= -179.99 &&
    lon <= 179.99 &&
    !isWithin50StateElevationCoverage(lat, lon);
}

async function fetchArcGISElevation(lat, lon) {
  if (!shouldUseArcGISElevationFallback(lat, lon)) return null;

  try {
    const url = `${ARCGIS_ELEVATION_API_URL}?lon=${encodeURIComponent(lon)}&lat=${encodeURIComponent(lat)}&relativeTo=meanSeaLevel&f=json&token=${encodeURIComponent(ARCGIS_ACCESS_TOKEN)}`;
    const response = await fetch(url);

    if (!response.ok) {
      console.warn("ArcGIS elevation returned:", response.status);
      return null;
    }

    const data = await response.json();
    if (data.error) {
      console.warn("ArcGIS elevation error:", data.error.message || data.error);
      return null;
    }

    const elevation = Number(data.result?.point?.z);
    return Number.isFinite(elevation) ? elevation : null;
  } catch (err) {
    console.error("ArcGIS elevation fetch error:", err);
    return null;
  }
}

async function fetchElevation(lat, lon) {
  try {
    const response = await fetch(
      `${ELEVATION_API_BASE}/api/v1/lookup?locations=${lat},${lon}`
    );

    if (!response.ok) {
      console.warn("Elevation server returned:", response.status);
      return fetchArcGISElevation(lat, lon);
    }

    const data = await response.json();
    const elevation = data.results?.[0]?.elevation ?? null;
    return elevation != null ? elevation : fetchArcGISElevation(lat, lon);
  } catch (err) {
    console.error("Elevation fetch error:", err);
    return fetchArcGISElevation(lat, lon);
  }
}

function getSortedUniqueRoutePoints(points) {
  const seenSequences = new Set();
  const seenFallbackKeys = new Set();

  return (Array.isArray(points) ? points : [])
    .filter(point =>
      Number.isFinite(point?.lat) &&
      Number.isFinite(point?.lon)
    )
    .slice()
    .sort((a, b) => {
      if (a.sequenceNum != null && b.sequenceNum != null) {
        return a.sequenceNum - b.sequenceNum;
      }

      return (a.timestamp || 0) - (b.timestamp || 0);
    })
    .filter(point => {
      if (point.sequenceNum != null) {
        if (seenSequences.has(point.sequenceNum)) return false;
        seenSequences.add(point.sequenceNum);
        return true;
      }

      const fallbackKey = `${point.timestamp || ""}:${point.lat}:${point.lon}`;
      if (seenFallbackKeys.has(fallbackKey)) return false;
      seenFallbackKeys.add(fallbackKey);
      return true;
    });
}

function buildSequenceLineSegments(points) {
  const segments = [];
  let currentSegment = [];
  let previousSequence = null;
  let previousTimestamp = null;
  let previousPoint = null;

  getSortedUniqueRoutePoints(points).forEach(point => {
    const pointCoords = [point.lon, point.lat];
    const sequence = point.sequenceNum;
    const timestamp = Number.isFinite(point.timestamp) ? point.timestamp : null;
    const distanceFromPrevious = previousPoint
      ? haversineDistance(previousPoint, point)
      : 0;

    if (
      currentSegment.length > 0 &&
      (
        (
          previousSequence != null &&
          sequence != null &&
          sequence - previousSequence > 1
        ) ||
        (
          previousTimestamp != null &&
          timestamp != null &&
          timestamp - previousTimestamp > NATIVE_PATH_BREAK_GAP_MS &&
          distanceFromPrevious > NATIVE_PATH_BREAK_DISTANCE_METERS
        ) ||
        distanceFromPrevious > ROUTE_RENDER_BREAK_DISTANCE_METERS
      )
    ) {
      if (currentSegment.length > 1) {
        segments.push(currentSegment);
      }

      currentSegment = [];
    }

    currentSegment.push(pointCoords);
    previousSequence = sequence;
    previousTimestamp = timestamp;
    previousPoint = point;
  });

  if (currentSegment.length > 1) {
    segments.push(currentSegment);
  }

  return segments;
}

function hasSequenceGap(points) {
  for (let i = 1; i < points.length; i++) {
    const previousSequence = points[i - 1].sequenceNum;
    const currentSequence = points[i].sequenceNum;
    const previousTimestamp = points[i - 1].timestamp;
    const currentTimestamp = points[i].timestamp;

    if (
      (
        previousSequence != null &&
        currentSequence != null &&
        currentSequence - previousSequence > 1
      ) ||
      (
        Number.isFinite(previousTimestamp) &&
        Number.isFinite(currentTimestamp) &&
        currentTimestamp - previousTimestamp > NATIVE_PATH_BREAK_GAP_MS &&
        haversineDistance(points[i - 1], points[i]) > NATIVE_PATH_BREAK_DISTANCE_METERS
      )
    ) {
      return true;
    }
  }

  return false;
}

function getLatestSequencePoint(points) {
  if (!points.length) return null;

  return points.reduce((latest, point) => {
    if (!latest) return point;

    if (point.sequenceNum != null && latest.sequenceNum != null) {
      return point.sequenceNum > latest.sequenceNum ? point : latest;
    }

    return (point.timestamp || 0) > (latest.timestamp || 0) ? point : latest;
  }, null);
}

async function fillMissingQueuedElevations(queue) {
  for (const point of queue.points) {
    if (point.elevation_m != null) continue;

    try {
      const elev = await fetchElevation(point.lat, point.lon);
      if (elev != null) {
        point.elevation_m = elev;
      }
    } catch (err) {
      console.warn("Could not fetch queued point elevation:", err);
    }
  }

  saveSyncQueue(queue);
}

async function flushSupabaseQueue() {
  if (isFlushingSupabase) return;
  if (!navigator.onLine) return;

let queue = loadSyncQueue();

const hasWork =
  queue.points.length > 0 ||
  Object.keys(queue.runUpdates).length > 0 ||
  Object.keys(queue.finishEvents).length > 0;

if (!hasWork) return;

isFlushingSupabase = true;

try {
    for (const [runId, updatePayload] of Object.entries(queue.runUpdates)) {
      const { error } = await supabaseClient
        .from("runs")
        .update(updatePayload)
        .eq("id", runId);

      if (error) {
        console.warn("Still unable to sync queued run update:", error);
        showConnectionStatus("Still offline - run summary is saved locally", "bad");
        saveSyncQueue(queue);
        return;
      }

      delete queue.runUpdates[runId];
      saveSyncQueue(queue);
    }

    for (const [runId, finishPayload] of Object.entries(queue.finishEvents)) {
      const { error } = await supabaseClient
        .from("runs")
        .update(finishPayload)
        .eq("id", runId);

      if (error) {
        console.warn("Still unable to sync finish event:", error);
        showConnectionStatus("Still offline - finish will sync later", "bad");
        saveSyncQueue(queue);
        return;
      }

      delete queue.finishEvents[runId];
      saveSyncQueue(queue);
    }

    if (runnerStoppedNativeHydrationInFlight) {
      return;
    }

    await fillMissingQueuedElevations(queue);
    queue = loadSyncQueue();
    while (queue.points.length > 0) {
      const pointPayload = queue.points[0];

      const { error } = await supabaseClient
        .from("run_points")
        .insert(pointPayload);

      if (error) {
        console.warn("Still unable to sync queued point:", error);
        showConnectionStatus("Still offline - run is saved locally", "bad");
        saveSyncQueue(queue);
        return;
      }

      queue.points.shift();
      saveSyncQueue(queue);
    }

    if (runnerSyncHadFailure) {
      runnerSyncHadFailure = false;
      showConnectionStatus("Connection restored - run synced", "good");
      hideConnectionStatusSoon();
    }
  } finally {
    isFlushingSupabase = false;
  }
}

  if (normalized.sequenceNum == null) return;

  const existingIndex = pathCoords.findIndex(existing =>
    existing.sequenceNum != null && existing.sequenceNum === normalized.sequenceNum
  );

  if (existingIndex >= 0) {
    pathCoords[existingIndex] = {
      ...pathCoords[existingIndex],
      ...normalized
    };
  } else {
    pathCoords.push(normalized);
  }
}

function appendRunnerNativeDisplayPoints(points) {

      point.sequenceNum != null && Number.isFinite(point.lat) && Number.isFinite(point.lon)
    )
    .forEach(point => {
      pathCoords.push(point);
    });
}

function applyRunnerNativeSnapshot(snapshot, includeFullPath = false, options = {}) {
  if (!snapshot) return false;

  const finalTimeMs = Number(options.finalTimeMs || 0);
  const focusLatestPoint = Boolean(options.focusLatestPoint);
  const appendOnly = Boolean(options.appendOnly);
  const renderPath = options.renderPath !== false;
  const updateSplits = options.updateSplits !== false;
  const points = Array.isArray(snapshot.points) ? snapshot.points : [];
  const latestRaw = snapshot.latestPoint && typeof snapshot.latestPoint === "object"
    ? snapshot.latestPoint
    : null;

  if (includeFullPath || pathCoords.length === 0 || !runnerNativeDisplayActive) {
    pathCoords = points.map(normalizeNativeRunPoint).filter(point =>
      point.sequenceNum != null && Number.isFinite(point.lat) && Number.isFinite(point.lon)
    );
  } else if (appendOnly) {
    appendRunnerNativeDisplayPoints(points);
  } else {
    points.forEach(upsertRunnerNativeDisplayPoint);
  }

  if (pathCoords.length > 0) {
    if (!appendOnly) {
      sortRunPoints(pathCoords);
    }
    if (renderPath) {
      addOrUpdatePath();
    }
    if (updateSplits) {
      updateSplitsPanel(pathCoords);
    }
    runnerNativeDisplayActive = true;
  }

  const latestPoint = latestRaw ? normalizeNativeRunPoint(latestRaw) : pathCoords[pathCoords.length - 1];
  if (!latestPoint) return points.length > 0;

  if (snapshot.firstRecordedAt) {
    const nativeStartMs = new Date(snapshot.firstRecordedAt).getTime();
    if (Number.isFinite(nativeStartMs)) {
      startTime = nativeStartMs;
    }
  } else if (!startTime && latestPoint.timestamp) {
    startTime = latestPoint.timestamp;
  }

  lastPointTime = latestPoint.timestamp || lastPointTime;
  totalDistanceMeters = latestPoint.distanceM ?? totalDistanceMeters;
  latestRunnerDistance = metersToMiles(totalDistanceMeters);
  latestRunnerPace = latestPoint.instPace ?? latestRunnerPace;
  latestRunnerAvgPace = latestPoint.avgPace ?? latestRunnerAvgPace;
  latestRunnerElevFt = latestPoint.elevationFt ?? latestRunnerElevFt;
  runnerNativeLastSequenceNum = Math.max(
    runnerNativeLastSequenceNum,
    snapshot.latestSequenceNum ?? latestPoint.sequenceNum ?? -1
  );

  if (latestPoint.sequenceNum != null) {
    pointSequence = Math.max(pointSequence, latestPoint.sequenceNum + 1);
  }

  const elapsedMs = finalTimeMs > 0
    ? finalTimeMs
    : startTime
      ? Math.max(0, (latestPoint.timestamp || Date.now()) - startTime)
      : 0;

  updateStatsDisplay(
    latestRunnerDistance,
    latestRunnerPace,
    latestRunnerAvgPace,
    elapsedMs,
    latestRunnerElevFt
  );

  if (
    isFollowing &&
    mapStyleReady &&
    map?.isStyleLoaded?.() &&
    Number.isFinite(latestPoint.lon) &&
    Number.isFinite(latestPoint.lat)
  ) {
    const latestCenter = [latestPoint.lon, latestPoint.lat];
    try {
      if (focusLatestPoint && map?.flyTo) {
        map.flyTo({
          center: latestCenter,
          zoom: Math.max(map.getZoom?.() || 0, 16),
          essential: true
        });
      } else {
        map.setCenter(latestCenter);
      }
    } catch (err) {
      console.warn("Native runner map recenter deferred:", err);
    }
  }

  return true;
}

async function refreshRunnerFromNativeSqlite(options = {}) {
  if (!isRunnerNativeSnapshotAllowed()) return false;

  const includeFullPath = Boolean(options.includeFullPath);
  const finalTimeMs = Number(options.finalTimeMs || 0);
  const limit = Number.isFinite(options.limit) ? options.limit : 0;
  const afterSequenceNum = includeFullPath
    ? -1
    : Number.isFinite(options.afterSequenceNum)
      ? options.afterSequenceNum
      : runnerNativeLastSequenceNum;
  const beforeSequenceNum = Number.isFinite(options.beforeSequenceNum)
    ? options.beforeSequenceNum
    : -1;

  try {
    const result = await LiveStrideNativeTracker.getNativeRunSnapshot({
      runId: activeRunId,
      afterSequenceNum,
      beforeSequenceNum,
      includeFullPath,
      limit
    });

    const snapshot = JSON.parse(result?.snapshotJson || "{}");
    return applyRunnerNativeSnapshot(snapshot, includeFullPath, {
      finalTimeMs,
      focusLatestPoint: Boolean(options.focusLatestPoint)
    });
  } catch (err) {
    console.warn("Native SQLite runner snapshot failed:", err);
    return false;
  }
}

async function hydrateRunnerFromNativeSqliteInBatches(options = {}) {
  if (!isRunnerNativeSnapshotAllowed()) return false;

  const finalTimeMs = Number(options.finalTimeMs || 0);
  const allowSplits = options.updateSplits !== false;
  const maxBatches = Number.isFinite(options.maxBatches)
    ? Math.max(1, Math.floor(options.maxBatches))
    : 40;
  let afterSequenceNum = Number.isFinite(options.afterSequenceNum)
    ? options.afterSequenceNum
    : -1;
  let loadedAny = false;
  let firstBatch = pathCoords.length === 0 && afterSequenceNum < 0;
  const hydrationStartedAt = performance.now?.() || Date.now();
  console.log("Native SQLite hydration started", {
    runId: activeRunId,
    afterSequenceNum,
    maxBatches,
    allowSplits
  });

  for (let i = 0; i < maxBatches; i++) {
    let snapshot;
    try {
      const result = await LiveStrideNativeTracker.getNativeRunSnapshot({
        runId: activeRunId,
        afterSequenceNum,
        beforeSequenceNum: -1,
        includeFullPath: false,
        limit: NATIVE_HYDRATION_BATCH_SIZE
      });
      snapshot = JSON.parse(result?.snapshotJson || "{}");
    } catch (err) {
      console.warn("Native SQLite batched hydration failed:", err);
      return loadedAny;
    }

    const points = Array.isArray(snapshot.points) ? snapshot.points : [];
    if (points.length === 0) break;

    const shouldRenderThisBatch = firstBatch
      || (i + 1) % NATIVE_HYDRATION_RENDER_EVERY_BATCHES === 0
      || points.length < NATIVE_HYDRATION_BATCH_SIZE;

    applyRunnerNativeSnapshot(snapshot, firstBatch, {
      finalTimeMs,
      focusLatestPoint: Boolean(options.focusLatestPoint) && firstBatch,
      appendOnly: !firstBatch,
      renderPath: shouldRenderThisBatch,
      updateSplits: allowSplits && shouldRenderThisBatch
    });
    loadedAny = true;
    firstBatch = false;

    const latestBatchPoint = points
      .map(normalizeNativeRunPoint)
      .filter(point => point.sequenceNum != null)
      .sort((a, b) => b.sequenceNum - a.sequenceNum)[0];

    if (!latestBatchPoint) break;
    afterSequenceNum = latestBatchPoint.sequenceNum;
    runnerNativeHydrationLoadedThroughSequenceNum = Math.max(
      runnerNativeHydrationLoadedThroughSequenceNum,
      latestBatchPoint.sequenceNum
    );

    if (points.length < NATIVE_HYDRATION_BATCH_SIZE) break;
    await yieldToBrowser();
  }

  if (loadedAny) {
    sortRunPoints(pathCoords);
    addOrUpdatePath();
    if (allowSplits) {
      updateSplitsPanel(pathCoords);
    }
  }

  const hydrationFinishedAt = performance.now?.() || Date.now();
  console.log("Native SQLite hydration finished", {
    runId: activeRunId,
    loadedAny,
    pointCount: pathCoords.length,
    loadedThroughSequence: runnerNativeHydrationLoadedThroughSequenceNum,
    durationMs: Math.round(hydrationFinishedAt - hydrationStartedAt)
  });

  return loadedAny;
}

  }
}

async function loadFullRunnerNativePathInBackground(finalTimeMs = 0) {
  if (!isRunnerNativeSnapshotAllowed()) return false;

  const fullStartedAt = performance.now?.() || Date.now();
  let afterSequenceNum = -1;
  let fullPoints = [];
  let latestSnapshot = null;

  console.log("Native SQLite full background hydration started", {
    runId: activeRunId
  });

  for (let i = 0; i < 120; i++) {
    let snapshot;
    try {
      const result = await LiveStrideNativeTracker.getNativeRunSnapshot({
        runId: activeRunId,
        afterSequenceNum,
        beforeSequenceNum: -1,
        includeFullPath: false,
        limit: NATIVE_HYDRATION_BATCH_SIZE
      });
      snapshot = JSON.parse(result?.snapshotJson || "{}");
    } catch (err) {
      console.warn("Native SQLite full background hydration failed:", err);
      return fullPoints.length > 0;
    }

    const points = Array.isArray(snapshot.points) ? snapshot.points : [];
    if (points.length === 0) break;

    fullPoints = fullPoints.concat(points);
    latestSnapshot = snapshot;

    const latestBatchPoint = points
      .map(normalizeNativeRunPoint)
      .filter(point => point.sequenceNum != null)
      .sort((a, b) => b.sequenceNum - a.sequenceNum)[0];

    if (!latestBatchPoint) break;
    afterSequenceNum = latestBatchPoint.sequenceNum;
    runnerNativeHydrationLoadedThroughSequenceNum = Math.max(
      runnerNativeHydrationLoadedThroughSequenceNum,
      latestBatchPoint.sequenceNum
    );

    if (points.length < NATIVE_HYDRATION_BATCH_SIZE) break;
    await new Promise(resolve => setTimeout(resolve, NATIVE_HYDRATION_BACKGROUND_DELAY_MS));
  }

  if (fullPoints.length === 0) return false;

  const fullSnapshot = {
    ...(latestSnapshot || {}),
    points: fullPoints
  };
  const hydrated = applyRunnerNativeSnapshot(fullSnapshot, true, {
    finalTimeMs,
    focusLatestPoint: false,
    renderPath: true,
    updateSplits: true
  });
  const fullFinishedAt = performance.now?.() || Date.now();

  console.log("Native SQLite full background hydration finished", {
    runId: activeRunId,
    hydrated,
    pointCount: fullPoints.length,
    durationMs: Math.round(fullFinishedAt - fullStartedAt)
  });

  return hydrated;
}

function continueRunnerNativeHydrationInBackground(finalTimeMs = 0) {
  if (!isRunnerNativeSnapshotAllowed()) return;

  setTimeout(() => {
    loadFullRunnerNativePathInBackground(finalTimeMs).finally(() => {
      runnerStoppedNativeHydrationInFlight = false;
    });
  }, NATIVE_HYDRATION_BACKGROUND_DELAY_MS);
}

function getSequenceGaps(points) {
  const sorted = points
    .filter(point => point.sequenceNum != null)
    .slice()
    .sort((a, b) => a.sequenceNum - b.sequenceNum);
  const gaps = [];

  for (let i = 1; i < sorted.length; i++) {
    const previousSequence = sorted[i - 1].sequenceNum;
    const currentSequence = sorted[i].sequenceNum;

    if (currentSequence - previousSequence > 1) {
      gaps.push({
        afterSequenceNum: previousSequence,
        beforeSequenceNum: currentSequence
      });
    }
  }

  return gaps;
}

async function loadRunnerPathFromSupabaseNative() {
  if (!activeRunId || !navigator.onLine) return false;

  try {
    const rows = await fetchAllNativeRunPoints(activeRunId);
    const points = rows
      .map(normalizeNativeRunPoint)
      .filter(point => point.sequenceNum != null && Number.isFinite(point.lat) && Number.isFinite(point.lon));

    if (points.length === 0) return false;

    pathCoords = points;
    sortRunPoints(pathCoords);
    addOrUpdatePath();
    updateSplitsPanel(pathCoords);
    runnerNativeDisplayActive = true;

    const latestPoint = getLatestSequencePoint(pathCoords) || pathCoords[pathCoords.length - 1];
    if (!latestPoint) return true;

    if (!startTime && pathCoords[0]?.timestamp) {
      startTime = pathCoords[0].timestamp;
    }

    lastPointTime = latestPoint.timestamp || lastPointTime;
    totalDistanceMeters = latestPoint.distanceM ?? totalDistanceMeters;
    latestRunnerDistance = metersToMiles(totalDistanceMeters);
    latestRunnerPace = latestPoint.instPace ?? latestRunnerPace;
    latestRunnerAvgPace = latestPoint.avgPace ?? latestRunnerAvgPace;
    latestRunnerElevFt = latestPoint.elevationFt ?? latestRunnerElevFt;
    runnerNativeLastSequenceNum = Math.max(runnerNativeLastSequenceNum, latestPoint.sequenceNum ?? -1);
    pointSequence = Math.max(pointSequence, (latestPoint.sequenceNum ?? -1) + 1);

    updateStatsDisplay(
      latestRunnerDistance,
      latestRunnerPace,
      latestRunnerAvgPace,
      startTime ? Math.max(0, (latestPoint.timestamp || Date.now()) - startTime) : 0,
      latestRunnerElevFt
    );

    if (isFollowing && Number.isFinite(latestPoint.lon) && Number.isFinite(latestPoint.lat)) {
      map.setCenter([latestPoint.lon, latestPoint.lat]);
    }

    return true;
  } catch (err) {
    console.warn("Supabase native runner wake path load failed:", err);
    return false;
  }
}

async function recoverRunnerPathOnWake() {
  if (!isRunnerNativeSnapshotAllowed()) return false;
  if (runnerWakeRecoveryInFlight) return false;

  runnerWakeRecoveryInFlight = true;

  try {
    const loadedFromNative = pathCoords.length === 0
      ? await hydrateRunnerFromNativeSqliteInBatches()
      : await refreshRunnerFromNativeSqlite({
          includeFullPath: false,
          limit: NATIVE_HYDRATION_BATCH_SIZE
        });

    if (loadedFromNative) {
      return true;
    }

    const loadedFromSupabase = await loadRunnerPathFromSupabaseNative();

    if (!loadedFromSupabase) {
      return false;
    }

    const supabaseLatestSequence = getLatestSequencePoint(pathCoords)?.sequenceNum ?? runnerNativeLastSequenceNum;
    const gaps = getSequenceGaps(pathCoords);

    for (const gap of gaps) {
      await refreshRunnerFromNativeSqlite({
        includeFullPath: false,
        afterSequenceNum: gap.afterSequenceNum,
        beforeSequenceNum: gap.beforeSequenceNum
      });
    }

    await refreshRunnerFromNativeSqlite({
      includeFullPath: false,
      afterSequenceNum: supabaseLatestSequence
    });

    return true;
  } finally {
    runnerWakeRecoveryInFlight = false;
  }
}

function ensureRunnerTimerRunning() {
  if (runnerTimerId !== null) return;

  runnerTimerId = setInterval(() => {
    if (!startTime) return;

    const elapsedMs = Date.now() - startTime;
    updateStatsDisplay(
      latestRunnerDistance,
      latestRunnerPace,
      latestRunnerAvgPace,
      elapsedMs,
      latestRunnerElevFt
    );
  }, 1000);
}

async function restoreRunnerFromNativeState() {
  if (!LiveStrideNativeTracker || document.body.classList.contains("viewer-mode")) {
    return false;
  }
  if (runnerNativeRestoreInFlight) {
    return false;
  }

  runnerNativeRestoreInFlight = true;

  try {
    const state = await LiveStrideNativeTracker.getNativeTrackerState();
    const nativeRunId = state?.runId;
    if (!nativeRunId) return false;

    const stoppedAtMs = Number(state.stoppedAtMs || 0);
    const isNativeRunning = state.running === true;
    const isNativeLocalOnly = state.localOnly === true;
    const stoppedRecently = stoppedAtMs > 0
      && Date.now() - stoppedAtMs <= STOPPED_NATIVE_RESTORE_GRACE_MS;

    if (!isNativeRunning && !stoppedRecently) {
      return false;
    }

    if (!isNativeRunning && stoppedAtMs > 0) {
      const restoredRunId = localStorage.getItem(RESTORED_STOPPED_NATIVE_RUN_KEY);
      if (restoredRunId === nativeRunId) {
        await clearNativeTrackerRestoreState();
        return false;
      }
    }

    if (!activeRunId) {
      activeRunId = nativeRunId;
    } else if (activeRunId !== nativeRunId) {
      return false;
    }

    const finalTimeMs = Number(state.finalTimeMs || 0);
    if (isNativeLocalOnly) {
      isPrivateMode = true;
      runnerName = "Private Run";
      if (broadcastCheckbox) broadcastCheckbox.checked = false;
    }
    statsBar.style.display = "block";

    let hydrated = false;
    if (!isNativeRunning && stoppedAtMs > 0) {
      runnerStoppedNativeHydrationInFlight = true;
      clearRunnerNativeRestoreDisplay();
      hydrated = await hydrateRunnerPreviewFromNativeSqlite(finalTimeMs);
      if (!hydrated) {
        hydrated = await hydrateRunnerFromNativeSqliteInBatches({
          finalTimeMs,
          focusLatestPoint: true,
          maxBatches: NATIVE_HYDRATION_FIRST_PAINT_BATCHES,
          updateSplits: false
        });
      }
      continueRunnerNativeHydrationInBackground(finalTimeMs);
    } else {
      hydrated = pathCoords.length === 0
        ? await hydrateRunnerFromNativeSqliteInBatches({ finalTimeMs, focusLatestPoint: true })
        : await refreshRunnerFromNativeSqlite({
            includeFullPath: false,
            finalTimeMs,
            focusLatestPoint: true,
            limit: NATIVE_HYDRATION_BATCH_SIZE
          });
    }

    if (!isNativeRunning && stoppedAtMs > 0) {
      if (hydrated) {
        localStorage.setItem(RESTORED_STOPPED_NATIVE_RUN_KEY, nativeRunId);
      }
      settleRunnerAfterNativeStop(finalTimeMs);
      if (isNativeLocalOnly) {
        exportGpxBtn.textContent = "Export GPX";
        exportGpxBtn.style.display = "inline-block";
        uploadStravaBtn.style.display = "none";
        exportUploadBtn.style.display = "none";
        showConnectionStatus("Private run stopped - GPX is ready to export", "good");
      } else {
        showConnectionStatus("Run stopped - background sync will finish if needed", "good");
      }
      hideConnectionStatusSoon();
    } else if (isNativeRunning) {
      startBtn.disabled = true;
      stopBtn.disabled = false;
      ensureRunnerTimerRunning();
      startRunnerNativeSnapshotPolling();
    }

    return hydrated;
  } catch (err) {
    console.warn("Native runner restore failed:", err);
    runnerStoppedNativeHydrationInFlight = false;
    return false;
  } finally {
    runnerNativeRestoreInFlight = false;
  }
}

function updateSupabaseViewerStats() {
  if (!pathCoords.length) return;

  let distMeters = 0;
  for (let i = 1; i < pathCoords.length; i++) {
    distMeters += haversineDistance(
      [pathCoords[i - 1].lon, pathCoords[i - 1].lat],
      [pathCoords[i].lon, pathCoords[i].lat]
    );
  }

  const firstTime = pathCoords[0].timestamp || Date.now();
  const lastTime = pathCoords[pathCoords.length - 1].timestamp || Date.now();
  const elapsedMs = Math.max(0, lastTime - firstTime);
  const distMiles = metersToMiles(distMeters);
  const avgPace = distMiles > 0 ? (elapsedMs / 60000) / distMiles : 0;
  const lastPoint = pathCoords[pathCoords.length - 1];

  const instPace = lastPoint.instPace ?? 0;

const displayElevationFt =
  lastPoint.elevationFt ??
  lastViewerData?.elevationFt ??
  latestRunnerElevFt ??
  null;

lastViewerData = {
  distance: distMiles,
  pace: instPace,
  avgPace,
  startTime: firstTime,
  current: [lastPoint.lon, lastPoint.lat],
  elevationFt: displayElevationFt
};

lastViewerElapsed = elapsedMs;

updateStatsDisplay(
  distMiles,
  instPace,
  avgPace,
  elapsedMs,
  displayElevationFt
);
}

function convertToGpx(runData) {
  const runDate = runData.startTime
    ? new Date(runData.startTime).toISOString().slice(0, 10)
    : new Date().toISOString().slice(0, 10);

  let gpx = `<?xml version="1.0" encoding="UTF-8"?>
<gpx creator="LiveStride" version="1.1" xmlns="http://www.topografix.com/GPX/1/1"
     xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
     xsi:schemaLocation="http://www.topografix.com/GPX/1/1
     http://www.topografix.com/GPX/1/1/gpx.xsd">
  <trk>
    <name>LiveStride Run ${runDate}</name>
    <trkseg>
`;

  runData.pathCoords.forEach(pt => {
    const ele = pt.elevation != null ? pt.elevation : 0;
    const time = pt.time || new Date(pt.timestamp).toISOString();
    gpx += `      <trkpt lat="${pt.lat}" lon="${pt.lon}">
        <ele>${ele}</ele>
        <time>${time}</time>
      </trkpt>
`;
  });

  gpx += `    </trkseg>
  </trk>
</gpx>`;

  return gpx;
}

}

async function shareGpx(filename, gpx, text = "Share your LiveStride GPX file.") {
  let shareUrl = null;

  if (Filesystem?.writeFile && Filesystem?.getUri) {
    try {
      await Filesystem.writeFile({
        path: filename,
        data: gpx,
        directory: "CACHE",
        encoding: "utf8"
      });
      const result = await Filesystem.getUri({
        directory: "CACHE",
        path: filename
      });
      shareUrl = result?.uri || null;
    } catch (err) {
      console.warn("Temporary GPX share file could not be created:", err);
    }
  }

  if (Share?.share) {
    try {
      await Share.share({
        title: "LiveStride GPX Export",
        text,
        url: shareUrl || undefined,
        dialogTitle: "Share your run GPX"
      });
    } catch (err) {
      console.warn("Share cancelled or failed:", err);
    }
  } else if (!shareUrl) {
    downloadGpxInBrowser(filename, gpx);
  }
}

function collectCoordinateLines(value, lines = []) {
  if (!Array.isArray(value) || value.length === 0) return lines;

  const isCoordinate = point =>
    Array.isArray(point) &&
    point.length >= 2 &&
    Number.isFinite(Number(point[0])) &&
    Number.isFinite(Number(point[1]));

  if (value.length >= 2 && value.every(isCoordinate)) {
    lines.push(value.map(point => [Number(point[0]), Number(point[1])]));
    return lines;
  }

  value.forEach(child => collectCoordinateLines(child, lines));
  return lines;
}

async function updateNativeMapMatchingContext(coord) {
  if (!activeRunId || !LiveStrideNativeTracker || !map?.getStyle) return;

  const now = Date.now();
  if (now - lastMapMatchingContextSentAt < 1500) return;

  try {
    const availableLayerIds = new Set(
      (map.getStyle().layers || []).map(layer => layer.id)
    );
    const pathLayers = [
      "Path",
      "Path minor",
      "livestride-roads-minor",
      "livestride-roads-major-casing",
      "livestride-roads-major"
    ].filter(id => availableLayerIds.has(id));
    if (pathLayers.length === 0) return;

    const centerPoint = map.project(coord);
    const radiusPixels = 100;
    const features = map.queryRenderedFeatures(
      [
        [centerPoint.x - radiusPixels, centerPoint.y - radiusPixels],
        [centerPoint.x + radiusPixels, centerPoint.y + radiusPixels]
      ],
      { layers: pathLayers }
    );

    const hasStructure = features.some(feature => {
      const brunnel = String(feature.properties?.brunnel || "").toLowerCase();
      return brunnel === "tunnel" || brunnel === "bridge";
    });
    if (!hasStructure) return;

    const segments = [];
    const seen = new Set();

    features.forEach((feature, featureIndex) => {
      const lines = collectCoordinateLines(feature.geometry?.coordinates);
      lines.forEach((points, lineIndex) => {
        if (points.length < 2) return;

        const baseId = feature.id ?? `rendered_${featureIndex}`;
        const id = `${baseId}:${lineIndex}`;
        const signature = `${id}:${JSON.stringify(points)}`;
        if (seen.has(signature)) return;
        seen.add(signature);

        segments.push({
          id,
          brunnel: String(feature.properties?.brunnel || "").toLowerCase(),
          subclass: String(feature.properties?.subclass || "").toLowerCase(),
          points
        });
      });
    });

    if (!segments.some(segment =>
      segment.brunnel === "tunnel" || segment.brunnel === "bridge"
    )) {
      return;
    }

    const contextSignature = JSON.stringify(segments);
    if (
      contextSignature === lastMapMatchingContextSignature &&
      now - lastMapMatchingContextSentAt < 10_000
    ) {
      return;
    }

    await LiveStrideNativeTracker.updateMapMatchingContext({
      runId: activeRunId,
      contextJson: JSON.stringify({ segments })
    });

    lastMapMatchingContextSignature = contextSignature;
    lastMapMatchingContextSentAt = now;
    console.log(
      "Updated native bridge/tunnel context:",
      segments.length,
      "path segments"
    );
  } catch (err) {
    console.warn("Bridge/tunnel context update skipped:", err);
  }
}

async function handleLocationUpdate(coord, accuracy = null, locationTime = Date.now()) {
  const now = typeof locationTime === "number"
    ? locationTime
    : new Date(locationTime).getTime();

  if (lastPointTime && now <= lastPointTime) {
    console.warn("Ignoring out-of-order location point:", new Date(now).toISOString());
    return;
  }

  if (isFollowing) {
    map.setCenter(coord);
  }
  updateNativeMapMatchingContext(coord);

  const reportedAccuracy = Number(accuracy);
  if (Number.isFinite(reportedAccuracy) && reportedAccuracy > 10) {
    console.warn("Ignoring inaccurate GPS point:", reportedAccuracy.toFixed(2), "m");
    return;
  }

  let elevMeters = null;
  let elevationFt = latestRunnerElevFt;

  let instPace = 0;
  let avgPace = 0;

  if (pathCoords.length > 0) {
    const prev = pathCoords[pathCoords.length - 1];
    const segmentMeters = haversineDistance([prev.lon, prev.lat], coord);
      const elapsedSeconds = lastPointTime ? (now - lastPointTime) / 1000 : 0;
  const speedMps = elapsedSeconds > 0 ? segmentMeters / elapsedSeconds : 0;
  const speedMph = speedMps * 2.23694;

  // Ignore duplicate / nearly stationary GPS noise
  if (segmentMeters < 1) {
    console.warn("Ignoring tiny GPS movement:", segmentMeters.toFixed(2), "m");
    return;
  }

  // Ignore impossible or bad timing points
  if (elapsedSeconds <= 0) {
    console.warn("Ignoring zero/negative time point:", elapsedSeconds);
    return;
  }

  // Ignore very slow drift that is probably GPS wobble, not running/walking
  if (speedMph < 1) {
    console.warn("Ignoring low-speed GPS drift:", speedMph.toFixed(2), "mph");
    return;
  }

  // Ignore unrealistic jumps
  if (speedMph > 140) {
    console.warn("Ignoring unrealistic GPS jump:", speedMph.toFixed(2), "mph");
    return;
  }
    totalDistanceMeters += segmentMeters;

    const segmentMiles = metersToMiles(segmentMeters);
    const elapsedMin = lastPointTime ? (now - lastPointTime) / 60000 : 0;
    instPace = segmentMiles > 0 ? elapsedMin / segmentMiles : 0;

    const totalMin = startTime ? (now - startTime) / 60000 : 0;
    const totalMiles = metersToMiles(totalDistanceMeters);
    avgPace = totalMiles > 0 ? totalMin / totalMiles : 0;
  } else {
    startTime = now;
  }

  lastPointTime = now;

  const recordedAt = new Date(now).toISOString();

  const newPoint = {
    lat: coord[1],
    lon: coord[0],
    timestamp: now,
    time: recordedAt,
    elevation: elevMeters,
    elevationFt,
    distance: metersToMiles(totalDistanceMeters),
    distanceM: totalDistanceMeters,
    instPace,
    avgPace
  };

  pathCoords.push(newPoint);

addOrUpdatePath();
updateSplitsPanel(pathCoords);

latestRunnerDistance = metersToMiles(totalDistanceMeters);
  latestRunnerPace = instPace;
  latestRunnerAvgPace = avgPace;
  latestRunnerElevFt = elevationFt;

if (broadcasting && broadcastRef) {
  broadcastRef.update({
    name: runnerName,
    run_id: activeRunId,
    finished: false,
    current: [coord[0], coord[1]],
    distance: latestRunnerDistance,
    pace: latestRunnerPace,
    avgPace: latestRunnerAvgPace,
    elevation: latestRunnerElevFt,
    timestamp: Date.now(),
    last_seen_at: new Date().toISOString(),
    startTime
  }).catch(err => {
    console.warn("Firebase lightweight summary update failed:", err);
  });
}

if (activeRunId) {
  const thisSequenceNum = pointSequence++;

  const pointPayload = {
    run_id: activeRunId,
    sequence_num: thisSequenceNum,
    lat: coord[1],
    lon: coord[0],
    accuracy: accuracy ?? null,
    elevation_m: elevMeters ?? null,
    recorded_at: recordedAt,
    inst_pace_min_per_mile: instPace,
    avg_pace_min_per_mile: avgPace,
    distance_m: totalDistanceMeters
  };

//await safeInsertRunPoint(pointPayload);
await safeUpdateRunSummary({
  total_distance_m: totalDistanceMeters,
  latest_lat: coord[1],
  latest_lon: coord[0],
  latest_timestamp: recordedAt
});

flushSupabaseQueue();

fetchElevation(coord[1], coord[0]).then(elev => {
  if (elev == null) return;

  elevMeters = elev;
  elevationFt = elev * 3.28084;

  newPoint.elevation = elevMeters;
  newPoint.elevationFt = elevationFt;
  latestRunnerElevFt = elevationFt;

  updateStatsDisplay(
    latestRunnerDistance,
    latestRunnerPace,
    latestRunnerAvgPace,
    Date.now() - startTime,
    latestRunnerElevFt
  );

  if (broadcasting && broadcastRef) {
    broadcastRef.update({
      elevation: latestRunnerElevFt,
      timestamp: Date.now(),
      last_seen_at: new Date().toISOString()
    }).catch(err => {
      console.warn("Firebase lightweight elevation update failed:", err);
    });
  }

  // Also update the queued/inserted payload object.
  // This matters if the point was queued before elevation returned.
  pointPayload.elevation_m = elevMeters;
updateQueuedPointElevation(activeRunId, thisSequenceNum, elevMeters);

// run_points elevation update disabled.

}).catch(err => {
  console.warn("Elevation update skipped:", err);
});
}
}

function normalizeRunPoint(pt) {
  return {
    sequenceNum: pt.sequence_num ?? null,
    lat: pt.lat,
    lon: pt.lon,
    timestamp: new Date(pt.recorded_at).getTime(),
    time: pt.recorded_at,
    elevation: pt.elevation_m,
    elevationFt: pt.elevation_m != null ? pt.elevation_m * 3.28084 : null,
    instPace: pt.inst_pace_min_per_mile ?? pt.instPace ?? 0,
    avgPace: pt.avg_pace_min_per_mile ?? pt.avgPace ?? 0,
    distanceM: pt.distance_m ?? pt.distanceM ?? null
  };
}

function sortRunPoints(points) {
  return points.sort((a, b) => {
    if (a.sequenceNum != null && b.sequenceNum != null) {
      return a.sequenceNum - b.sequenceNum;
    }

    return (a.timestamp || 0) - (b.timestamp || 0);
  });
}

function upsertSortedRunPoint(pt) {
  const normalized = normalizeRunPoint(pt);

  const existingIndex = pathCoords.findIndex(p =>
    p.sequenceNum != null &&
    normalized.sequenceNum != null &&
    p.sequenceNum === normalized.sequenceNum
  );

  if (existingIndex >= 0) {
    pathCoords[existingIndex] = {
      ...pathCoords[existingIndex],
      ...normalized
    };
  } else {
    pathCoords.push(normalized);
  }

  sortRunPoints(pathCoords);
}

function normalizeNativeRunPoint(pt) {
  const elevationM = pt.elevation_m ?? pt.elevation ?? null;
  const recordedAt = pt.recorded_at ?? pt.time;

  return {
    sequenceNum: pt.sequence_num ?? pt.sequenceNum ?? null,
    lat: pt.lat,
    lon: pt.lon,
    timestamp: recordedAt ? new Date(recordedAt).getTime() : pt.timestamp,
    time: recordedAt,
    elevation: elevationM,
    elevationFt: elevationM != null ? elevationM * 3.28084 : null,
    instPace: pt.inst_pace_min_per_mile ?? pt.instPace ?? 0,
    avgPace: pt.avg_pace_min_per_mile ?? pt.avgPace ?? 0,
    distanceM: pt.distance_m ?? pt.distanceM ?? null
  };
}

function upsertSortedNativeRunPoint(pt) {
  const normalized = normalizeNativeRunPoint(pt);

  const existingIndex = nativePathCoords.findIndex(p =>
    p.sequenceNum != null &&
    normalized.sequenceNum != null &&
    p.sequenceNum === normalized.sequenceNum
  );

  if (existingIndex >= 0) {
    nativePathCoords[existingIndex] = {
      ...nativePathCoords[existingIndex],
      ...normalized
    };
  } else {
    nativePathCoords.push(normalized);
  }

  nativePathCoords.sort((a, b) => {
    if (a.sequenceNum != null && b.sequenceNum != null) {
      return a.sequenceNum - b.sequenceNum;
    }

    return (a.timestamp || 0) - (b.timestamp || 0);
  });
}

function updateViewerFromNativePoint(pt) {
  const nativePoint = normalizeNativeRunPoint
    ? normalizeNativeRunPoint(pt)
    : {
        sequenceNum: pt.sequence_num ?? null,
        lat: pt.lat,
        lon: pt.lon,
        timestamp: new Date(pt.recorded_at).getTime(),
        time: pt.recorded_at,
        elevation: pt.elevation_m,
        elevationFt: pt.elevation_m != null ? pt.elevation_m * 3.28084 : null,
        instPace: pt.inst_pace_min_per_mile ?? 0,
        avgPace: pt.avg_pace_min_per_mile ?? 0,
        distanceM: pt.distance_m ?? null
      };

  const nativeStartTime = nativePathCoords.length > 0
    ? nativePathCoords[0].timestamp
    : nativePoint.timestamp;

  const elapsedMs = Math.max(0, nativePoint.timestamp - nativeStartTime);
  const distanceMi = nativePoint.distanceM != null
    ? metersToMiles(nativePoint.distanceM)
    : 0;

  const elevationFt =
    nativePoint.elevationFt ??
    lastViewerData?.elevationFt ??
    latestRunnerElevFt ??
    null;

  lastViewerData = {
    distance: distanceMi,
    pace: nativePoint.instPace ?? 0,
    avgPace: nativePoint.avgPace ?? 0,
    startTime: nativeStartTime,
    current: [nativePoint.lon, nativePoint.lat],
    elevationFt
  };

  lastViewerElapsed = elapsedMs;

  updateStatsDisplay(
    distanceMi,
    nativePoint.instPace ?? 0,
    nativePoint.avgPace ?? 0,
    elapsedMs,
    elevationFt
  );

  if (!viewerMarker) {
    viewerMarker = new maplibregl.Marker({ color: "red" })
      .setLngLat([nativePoint.lon, nativePoint.lat])
      .addTo(map);
  } else {
    viewerMarker.setLngLat([nativePoint.lon, nativePoint.lat]);
  }

  if (viewerFollowing) {
    map.flyTo({
      center: [nativePoint.lon, nativePoint.lat],
      zoom: 16
    });
  }
}

async function fetchAllNativeRunPoints(runId) {
  const pageSize = 1000;
  let from = 0;
  const allPoints = [];

  while (true) {
    const to = from + pageSize - 1;
    const { data, error } = await supabaseClient
      .from("run_points_native_test")
      .select("*")
      .eq("run_id", runId)
      .order("sequence_num", { ascending: true })
      .range(from, to);

    if (error) throw error;

    const page = data || [];
    allPoints.push(...page);

    if (page.length < pageSize) break;
    from += pageSize;
  }

  return allPoints;
}
async function startNativePathComparison(runId) {
  if (!runId) return;

  if (activeNativeSupabaseChannel) {
    supabaseClient.removeChannel(activeNativeSupabaseChannel);
    activeNativeSupabaseChannel = null;
  }

  nativePathCoords = [];

  // Only show the native purple path in viewer mode.
  if (document.body.classList.contains("viewer-mode")) {
    clearNativeRunPathLayer();
  }

  let data;

  try {
    data = await fetchAllNativeRunPoints(runId);
  } catch (error) {
    console.error("Initial native path load error:", error);
    return;
  }

  nativePathCoords = data
    .map(normalizeNativeRunPoint)
    .sort((a, b) => {
      if (a.sequenceNum != null && b.sequenceNum != null) {
        return a.sequenceNum - b.sequenceNum;
      }
      return (a.timestamp || 0) - (b.timestamp || 0);
    });
  nativeRouteGapActive = hasSequenceGap(nativePathCoords);

  if (document.body.classList.contains("viewer-mode")) {
    addOrUpdateNativePath();
    updateSplitsPanel(nativePathCoords);

    if (nativePathCoords.length > 0) {
      const lastNativePoint = getLatestSequencePoint(nativePathCoords);

      updateViewerFromNativePoint(lastNativePoint);
    }
  }

  activeNativeSupabaseChannel = supabaseClient
    .channel(`native_run_${runId.replace(/-/g, "_")}`)
    .on(
      "postgres_changes",
      {
        event: "INSERT",
        schema: "public",
        table: "run_points_native_test",
        filter: `run_id=eq.${runId}`
      },
      payload => {
        const pt = payload.new;
        const previousLatestPoint = getLatestSequencePoint(nativePathCoords);
        const incomingPoint = normalizeNativeRunPoint(pt);
        const incomingSequence = incomingPoint.sequenceNum;
        const previousLatestSequence = previousLatestPoint?.sequenceNum ?? null;
        const wasConnectionLost = viewerConnectionLost;
        const wasConnectionWarning = viewerConnectionWarning;

        upsertSortedNativeRunPoint(pt);
        lastPointReceivedTime = Date.now();
        viewerConnectionWarning = false;

        if (viewerConnectionLost) {
          viewerConnectionLost = false;
        }

        const hadGap = nativeRouteGapActive;
        const hasGap = hasSequenceGap(nativePathCoords);
        nativeRouteGapActive = hasGap;

        const incomingSequenceGap =
          incomingSequence != null &&
          previousLatestSequence != null &&
          incomingSequence - previousLatestSequence > 1;
        const cleanSequentialResume =
          incomingSequence != null &&
          previousLatestSequence != null &&
          incomingSequence - previousLatestSequence === 1;

        if (incomingSequenceGap) {
          showViewerStatus("Reconnecting - rebuilding route", "good");
          hideViewerStatusSoon("Reconnecting - rebuilding route", 6000);
        } else if (hadGap && !hasGap) {
          showViewerStatus("Route restored", "good");
          hideViewerStatusSoon("Route restored", 3000);
        } else if ((wasConnectionLost || wasConnectionWarning) && cleanSequentialResume) {
          showViewerStatus("Motion detected", "good");
          hideViewerStatusSoon("Motion detected", 3000);
        } else if (!hasGap && (wasConnectionLost || wasConnectionWarning)) {
          showViewerStatus("Runner reconnected", "good");
          hideViewerStatusSoon("Runner reconnected", 3000);
        }

        // Only render purple native path and red marker in viewer mode.
        if (document.body.classList.contains("viewer-mode")) {
          addOrUpdateNativePath();
          updateSplitsPanel(nativePathCoords);
          const latestNativePoint = getLatestSequencePoint(nativePathCoords);
          if (latestNativePoint) {
            updateViewerFromNativePoint(latestNativePoint);
          }
        }
      }
    )
    .subscribe();
}

function trackRunner(key) {
  currentViewerRunIsLive = null;

  if (activeSupabaseChannel) {
    supabaseClient.removeChannel(activeSupabaseChannel);
    activeSupabaseChannel = null;
  }

 if (activeNativeSupabaseChannel) {
  supabaseClient.removeChannel(activeNativeSupabaseChannel);

  if (activeListener) {
    activeListener.off();

    clearInterval(viewerTimerId);
    viewerTimerId = null;
  }

    if (!data || !data.current) return;

    if (!viewerMarker) {
      viewerMarker = new maplibregl.Marker({ color: "red" })
        .setLngLat(data.current)

      lat: pt.lat,
      lon: pt.lon,

    lastViewerData = {

    };

      ? data.timestamp - data.startTime
      : lastViewerElapsed;

        lastViewerData.current[1],

    updateStatsDisplay(

      lastViewerData.avgPace,

    );

    if (data.finished && viewerTimerId !== null) {
      clearInterval(viewerTimerId);
      viewerTimerId = null;
    }

    if (viewerFollowing) {
      map.flyTo({ center: data.current, zoom: 16 });
    }
  });

  if (viewerTimerId === null) {
    viewerTimerId = setInterval(async () => {
      const nowData = (await ref.once("value")).val();
      if (nowData && nowData.finished) {
        clearInterval(viewerTimerId);
        viewerTimerId = null;
        return;
      }

      if (!lastViewerData) return;

      lastViewerElapsed += 1000;
      updateStatsDisplay(
        lastViewerData.distance,
        lastViewerData.pace,
        lastViewerData.avgPace,
        lastViewerElapsed,
        lastViewerData.elevationFt ?? null
      );
    }, 1000);
  }
}

layerSelect.addEventListener("change", () => {
  const selectedStyleKey = layerSelect.value === "liveStride" && !liveStridePmtilesReady
    ? "satellite"
    : layerSelect.value;
  const newStyle = STYLES[selectedStyleKey] || STYLES[DEFAULT_STYLE_KEY];
  updateMapStyleBodyClass(selectedStyleKey);
  mapStyleReady = false;
  runPathRenderPending = pathCoords.length > 0 || runPathRenderPending;
  nativePathRenderPending = nativePathCoords.length > 0 || nativePathRenderPending;
  map.setStyle(newStyle);
});

}, 0);
